# 04-长任务稳定性

长任务执行面临诸多挑战：执行时间过长、中间失败、上下文丢失等。本文档详细介绍保持长任务稳定性的技术方案。

## 长任务挑战概述

## 断点续做机制

### 检查点系统

```python
import json
import os
from pathlib import Path
from datetime import datetime
from typing import Any, Optional

@dataclass
class Checkpoint:
    """检查点"""
    id: str
    task_id: str
    step_index: int
    state_data: dict
    created_at: datetime

class CheckpointManager:
    """检查点管理器"""

    def __init__(self, checkpoint_dir: str):
        self.checkpoint_dir = Path(checkpoint_dir)
        self.checkpoint_dir.mkdir(parents=True, exist_ok=True)

    def save_checkpoint(self, checkpoint: Checkpoint):
        """保存检查点"""

        file_path = self.checkpoint_dir / f"{checkpoint.task_id}.json"

        data = {
            "id": checkpoint.id,
            "task_id": checkpoint.task_id,
            "step_index": checkpoint.step_index,
            "state_data": checkpoint.state_data,
            "created_at": checkpoint.created_at.isoformat()
        }

        with open(file_path, 'w', encoding='utf-8') as f:
            json.dump(data, f, ensure_ascii=False, indent=2)

    def load_checkpoint(self, task_id: str) -> Optional[Checkpoint]:
        """加载检查点"""

        file_path = self.checkpoint_dir / f"{task_id}.json"

        if not file_path.exists():
            return None

        with open(file_path, 'r', encoding='utf-8') as f:
            data = json.load(f)

        return Checkpoint(
            id=data["id"],
            task_id=data["task_id"],
            step_index=data["step_index"],
            state_data=data["state_data"],
            created_at=datetime.fromisoformat(data["created_at"])
        )

    def list_checkpoints(self) -> list:
        """列出所有检查点"""
        checkpoints = []

        for file in self.checkpoint_dir.glob("*.json"):
            with open(file, 'r') as f:
                data = json.load(f)
                checkpoints.append({
                    "task_id": data["task_id"],
                    "step_index": data["step_index"],
                    "created_at": data["created_at"]
                })

        return checkpoints

    def delete_checkpoint(self, task_id: str):
        """删除检查点"""
        file_path = self.checkpoint_dir / f"{task_id}.json"
        if file_path.exists():
            file_path.unlink()
```

### 自动保存检查点

```python
class CheckpointableExecutor:
    """支持检查点的执行器"""

    def __init__(self, executor, checkpoint_manager: CheckpointManager):
        self.executor = executor
        self.checkpoints = checkpoint_manager

    async def execute_with_checkpoint(self, task: dict,
                                     task_id: str,
                                     checkpoint_interval: int = 5) -> Any:
        """带检查点执行"""

        # 尝试加载之前的检查点
        checkpoint = self.checkpoints.load_checkpoint(task_id)
        start_step = checkpoint.step_index if checkpoint else 0

        # 执行任务
        step_results = []

        for i, step in enumerate(task["steps"]):
            # 从检查点恢复
            if i < start_step:
                # 跳过已完成的步骤
                if checkpoint and i in checkpoint.state_data.get("results", {}):
                    step_results.append(checkpoint.state_data["results"][i])
                    continue

            print(f"执行步骤 {i + 1}/{len(task['steps'])}")

            # 执行步骤
            result = await self.executor.execute_step(step, step_results)
            step_results.append(result)

            # 定期保存检查点
            if (i + 1) % checkpoint_interval == 0:
                self._save_checkpoint(task_id, i, step_results, task)

            # 检查是否有严重错误需要回退
            if self._is_fatal_error(result):
                # 回退到上一个检查点
                return await self._rollback(task_id, task, step_results)

        # 清理检查点
        self.checkpoints.delete_checkpoint(task_id)

        return step_results

    def _save_checkpoint(self, task_id: str, step_index: int,
                        results: list, original_task: dict):
        """保存检查点"""

        checkpoint = Checkpoint(
            id=f"cp_{task_id}_{step_index}",
            task_id=task_id,
            step_index=step_index,
            state_data={
                "results": results,
                "original_task": original_task
            },
            created_at=datetime.now()
        )

        self.checkpoints.save_checkpoint(checkpoint)
        print(f"检查点已保存: step {step_index}")

    def _is_fatal_error(self, result: Any) -> bool:
        """判断是否为致命错误"""
        if isinstance(result, dict):
            return result.get("fatal", False)
        return False

    async def _rollback(self, task_id: str, task: dict,
                       results: list) -> Any:
        """回退到上一个检查点"""

        checkpoint = self.checkpoints.load_checkpoint(task_id)

        if checkpoint:
            print(f"回退到检查点 step {checkpoint.step_index}")
            # 从检查点重新执行
            return await self.execute_with_checkpoint(
                task, task_id, checkpoint_interval=999
            )
        else:
            print("无法回退：无检查点")
            return None
```

## 错误恢复机制

### 重试策略

```python
import asyncio
from typing import Callable, Any
from functools import wraps

class RetryStrategy:
    """重试策略"""

    def __init__(self,
                 max_attempts: int = 3,
                 base_delay: float = 1.0,
                 max_delay: float = 60.0,
                 exponential_base: float = 2.0,
                 retriable_errors: list = None):
        self.max_attempts = max_attempts
        self.base_delay = base_delay
        self.max_delay = max_delay
        self.exponential_base = exponential_base
        self.retriable_errors = retriable_errors or [
            "timeout", "connection", "rate_limit"
        ]

    def is_retriable(self, error: Exception) -> bool:
        """判断错误是否可重试"""
        error_msg = str(error).lower()
        return any(retriable in error_msg for retriable in self.retriable_errors)

    def get_delay(self, attempt: int) -> float:
        """计算延迟"""
        delay = self.base_delay * (self.exponential_base ** attempt)
        return min(delay, self.max_delay)

async def with_retry(func: Callable, strategy: RetryStrategy,
                    *args, **kwargs) -> Any:
    """带重试的函数执行"""

    last_error = None

    for attempt in range(strategy.max_attempts):
        try:
            return await func(*args, **kwargs)

        except Exception as e:
            last_error = e

            if not strategy.is_retriable(e):
                raise

            if attempt < strategy.max_attempts - 1:
                delay = strategy.get_delay(attempt)
                print(f"重试 {attempt + 1}/{strategy.max_attempts}, "
                      f"等待 {delay}s: {e}")
                await asyncio.sleep(delay)

    raise last_error
```

### 错误恢复流程

```python
class ErrorRecovery:
    """错误恢复机制"""

    def __init__(self, executor):
        self.executor = executor

    async def execute_with_recovery(self, task: dict) -> dict:
        """带错误恢复的执行"""

        results = []
        errors = []

        for i, step in enumerate(task["steps"]):
            try:
                # 尝试执行
                result = await with_retry(
                    self.executor.execute_step,
                    RetryStrategy(max_attempts=3),
                    step
                )
                results.append({"success": True, "result": result})

            except Exception as e:
                # 记录错误
                errors.append({
                    "step": i,
                    "error": str(e),
                    "timestamp": datetime.now().isoformat()
                })

                # 尝试恢复
                recovery_result = await self._attempt_recovery(
                    step, e, results
                )

                if recovery_result:
                    results.append(recovery_result)
                else:
                    # 无法恢复，停止执行
                    break

        return {
            "results": results,
            "errors": errors,
            "completed": len(results) == len(task["steps"])
        }

    async def _attempt_recovery(self, failed_step: dict,
                                error: Exception,
                                previous_results: list) -> Optional[dict]:
        """尝试恢复"""

        error_type = self._classify_error(error)

        if error_type == "timeout":
            # 超时：重试一次
            return None  # 让外层重试

        elif error_type == "resource":
            # 资源不足：等待后重试
            await asyncio.sleep(5)
            return None

        elif error_type == "validation":
            # 验证错误：尝试修正
            return await self._fix_and_retry(failed_step, error)

        elif error_type == "permission":
            # 权限错误：无法恢复
            return None

        else:
            # 未知错误：记录并继续
            return {
                "success": False,
                "error": str(error),
                "recovered": False
            }

    def _classify_error(self, error: Exception) -> str:
        """分类错误"""
        error_msg = str(error).lower()

        if "timeout" in error_msg:
            return "timeout"
        elif "memory" in error_msg or "cpu" in error_msg:
            return "resource"
        elif "validation" in error_msg or "invalid" in error_msg:
            return "validation"
        elif "permission" in error_msg or "denied" in error_msg:
            return "permission"
        else:
            return "unknown"

    async def _fix_and_retry(self, step: dict, error: Exception) -> Optional[dict]:
        """尝试修复并重试"""
```

## 上下文管理

### 长期记忆保持

```python
class LongTermContextManager:
    """长期上下文管理器"""

    def __init__(self, max_tokens: int = 80000):
        self.max_tokens = max_tokens
        self.core_memory = []  # 核心记忆（永远保留）
        self.summary_memory = []  # 摘要记忆
        self.full_history = []  # 完整历史

    def add_interaction(self, role: str, content: str, token_count: int):
        """添加交互"""

        message = {
            "role": role,
            "content": content,
            "tokens": token_count,
            "timestamp": datetime.now()
        }

        self.full_history.append(message)

        # 定期生成摘要
        if len(self.full_history) % 10 == 0:
            self._generate_summary()

    def get_context(self, max_tokens: int = None) -> list:
        """获取当前上下文"""

        max_tokens = max_tokens or self.max_tokens

        # 1. 核心记忆
        context = list(self.core_memory)

        # 2. 计算剩余空间
        used_tokens = sum(m["tokens"] for m in context)
        remaining = max_tokens - used_tokens

        # 3. 添加摘要
        for summary in self.summary_memory[-3:]:
            if remaining >= summary["tokens"]:
                context.append({
                    "role": "system",
                    "content": f"之前对话摘要: {summary['content']}"
                })
                remaining -= summary["tokens"]

        # 4. 添加最近的完整历史
        for msg in reversed(self.full_history[-10:]):
            if remaining >= msg["tokens"]:
                context.insert(0, msg)
                remaining -= msg["tokens"]
            else:
                break

        return context

    def _generate_summary(self):
        """生成摘要"""

        if not self.full_history:
            return

        # 取最近的 20 条进行摘要
        recent = self.full_history[-20:]
        content = "\n".join(
            f"{m['role']}: {m['content'][:100]}"
            for m in recent
        )

        # 这里调用 LLM 生成摘要
        summary = f"摘要: 共 {len(recent)} 条对话，主要讨论..."

        self.summary_memory.append({
            "content": summary,
            "tokens": len(summary.split()) * 1.3,
            "timestamp": datetime.now()
        })

        # 清理过旧的完整历史
        self.full_history = self.full_history[-30:]
```

### 进度追踪

```python
class ProgressTracker:
    """进度追踪器"""

    def __init__(self, total_steps: int):
        self.total_steps = total_steps
        self.current_step = 0
        self.step_results = []
        self.checkpoints = {}

    def update_progress(self, step: int, result: dict):
        """更新进度"""

        self.current_step = step
        self.step_results.append({
            "step": step,
            "result": result,
            "timestamp": datetime.now()
        })

    def get_progress(self) -> dict:
        """获取进度"""

        return {
            "current": self.current_step,
            "total": self.total_steps,
            "percentage": (self.current_step / self.total_steps * 100)
                         if self.total_steps > 0 else 0,
            "completed": self.current_step >= self.total_steps
        }

    def save_checkpoint(self):
        """保存进度检查点"""
        self.checkpoints[self.current_step] = {
            "step_results": list(self.step_results),
            "timestamp": datetime.now()
        }

    def restore_checkpoint(self, step: int):
        """恢复检查点"""

        if step in self.checkpoints:
            checkpoint = self.checkpoints[step]
            self.step_results = checkpoint["step_results"]
            self.current_step = step
            return True

        return False
```

```python
class LongTaskRunner:
    """长任务执行器"""

    def __init__(self, config: dict):
        self.checkpoint_manager = CheckpointManager(
            config.get("checkpoint_dir", "./checkpoints")
        )
        self.executor = CheckpointableExecutor(
            DefaultExecutor(), self.checkpoint_manager
        )
        self.error_recovery = ErrorRecovery(self.executor)
        self.context_manager = LongTermContextManager(
            config.get("max_tokens", 80000)
        )
        self.progress = None

    async def run_long_task(self, task: dict, task_id: str) -> dict:
        """运行长任务"""

        print(f"开始长任务: {task_id}")
        self.progress = ProgressTracker(len(task["steps"]))

        # 检查是否有之前的检查点
        checkpoint = self.checkpoint_manager.load_checkpoint(task_id)
        if checkpoint:
            print(f"找到检查点，从 step {checkpoint.step_index} 恢复")

        # 执行任务
        try:
            result = await self._execute_with_stability(task, task_id)

            # 成功，清理检查点
            self.checkpoint_manager.delete_checkpoint(task_id)

            return {
                "status": "success",
                "result": result,
                "progress": self.progress.get_progress()
            }

        except Exception as e:
            return {
                "status": "failed",
                "error": str(e),
                "progress": self.progress.get_progress()
            }

    async def _execute_with_stability(self, task: dict, task_id: str) -> Any:
        """稳定执行"""

        for i, step in enumerate(task["steps"]):
            # 更新进度
            self.progress.update_progress(i, {"status": "executing"})

            # 定期保存检查点
            if i % 5 == 0:
                self.progress.save_checkpoint()

            # 执行
            result = await self.executor.execute_with_checkpoint(
                task, task_id, checkpoint_interval=5
            )

            # 更新上下文
            self.context_manager.add_interaction(
                "assistant",
                f"执行步骤 {i + 1}: {result}",
                len(str(result).split()) * 1.3
            )

            # 检查是否完成
            if i >= len(task["steps"]) - 1:
                break

        return result

    def get_current_status(self) -> dict:
        """获取当前状态"""

        return {
            "progress": self.progress.get_progress() if self.progress else None,
            "context_tokens": sum(
                m["tokens"] for m in self.context_manager.full_history
            )
        }
```
