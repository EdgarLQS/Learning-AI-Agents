# 02-任务拆解方法

任务拆解是让 Agent 能够处理复杂问题的核心技术。本文档详细介绍各种任务拆解方法和策略。

## 常见拆解策略

### 1. 自顶向下拆解 (Top-Down)

```python
class TopDownDecomposer:
    """自顶向下拆解器"""

    def __init__(self, llm_client):
        self.llm = llm_client

    async def decompose(self, task: str, context: dict) -> List[dict]:
        """自顶向下拆解任务"""

        # 第一层：主要阶段
        stages = await self._decompose_into_stages(task)

        # 第二层：具体任务
        tasks = []
        for stage in stages:
            subtasks = await self._decompose_stage(stage, context)
            tasks.extend(subtasks)

        return tasks

    async def _decompose_into_stages(self, task: str) -> List[dict]:
        """拆解为主要阶段"""

        prompt = f"""将以下任务拆解为 3-6 个主要阶段：

任务：{task}

返回 JSON 格式：
{{
  "stages": [
    {{"id": 1, "name": "阶段名称", "description": "阶段描述"}},
    ...
  ]
}}"""

        # 调用 LLM
        # ...

        return [{"id": 1, "name": "分析", "description": "分析现有代码"},
                {"id": 2, "name": "设计", "description": "设计解决方案"},
                {"id": 3, "name": "实现", "description": "实现代码变更"},
                {"id": 4, "name": "验证", "description": "验证修改正确"}]

    async def _decompose_stage(self, stage: dict, context: dict) -> List[dict]:
        """拆解阶段为具体任务"""

        prompt = f"""将以下阶段拆解为具体的可执行任务：

阶段：{stage['name']}
描述：{stage['description']}

上下文：
{self._format_context(context)}

每个任务应该：
1. 明确具体（可以在 1-2 分钟内理解要做什么）
2. 有明确的完成标准
3. 可以独立验证

返回 JSON 格式：
{{
  "tasks": [
    {{"id": "stage_id_1", "description": "...", "depends_on": []}},
    ...
  ]
}}"""

        return []
```

### 2. 分而治之 (Divide and Conquer)

```python
class DivideConquerDecomposer:
    """分而治之拆解器"""

    def __init__(self, llm_client):
        self.llm = llm_client

    async def decompose(self, problem: str, solution_space: dict) -> List[dict]:
        """分而治之拆解"""

        # 1. 分解问题
        sub_problems = await self._split_problem(problem)

        # 2. 对每个子问题递归拆解
        all_tasks = []
        for sub in sub_problems:
            tasks = await self._recursive_decompose(sub, solution_space)
            all_tasks.extend(tasks)

        # 3. 处理合并
        final_tasks = self._merge_tasks(all_tasks)

        return final_tasks

    async def _split_problem(self, problem: str) -> List[str]:
        """将问题拆分为子问题"""

        # 可以基于维度拆分：文件、模块、功能、场景
        return []

    async def _recursive_decompose(self, sub_problem: str,
                                   solution_space: dict,
                                   depth: int = 0) -> List[dict]:
        """递归拆解"""

        if depth > 3:  # 深度限制
            return [{"description": sub_problem, "depth": depth}]

        # 判断是否需要继续拆分
        if self._needs_splitting(sub_problem, solution_space):
            children = await self._split_problem(sub_problem)
            tasks = []
            for child in children:
                tasks.extend(await self._recursive_decompose(
                    child, solution_space, depth + 1
                ))
            return tasks
        else:
            return [{"description": sub_problem, "depth": depth}]

    def _needs_splitting(self, problem: str, solution_space: dict) -> bool:
        """判断是否需要继续拆分"""
        # 基于问题的复杂度和当前深度
        complex_keywords = ["多个", "重构", "迁移", "全面"]
        return any(kw in problem for kw in complex_keywords)

    def _merge_tasks(self, tasks: List[dict]) -> List[dict]:
        """合并任务"""
        # 去重、排序、合并相似任务
        return tasks
```

```python
class TemplateBasedDecomposer:
    """基于模板的任务拆解器"""

    # 常见任务模板
    TEMPLATES = {
        "修改文件": [

            {"step": "进行修改", "tool": "edit_file"},
            {"step": "验证修改", "tool": "test"}
        ],
        "添加功能": [
            {"step": "理解现有结构", "tool": "read_file"},
            {"step": "设计实现方案", "tool": "think"},
            {"step": "实现功能", "tool": "write_file"},
            {"step": "添加测试", "tool": "write_file"},
            {"step": "运行测试", "tool": "test"}
        ],
        "修复bug": [
            {"step": "复现问题", "tool": "analyze"},
            {"step": "定位根因", "tool": "search"},
            {"step": "修复问题", "tool": "edit_file"},
            {"step": "验证修复", "tool": "test"}
        ],
        "代码审查": [
            {"step": "获取代码", "tool": "read_file"},
            {"step": "分析代码", "tool": "analyze"},
            {"step": "提出建议", "tool": "think"}
        ],
        "重构": [
            {"step": "分析现有代码", "tool": "read_file"},
            {"step": "设计新结构", "tool": "think"},
            {"step": "逐步迁移", "tool": "edit_file"},
            {"step": "保持功能兼容", "tool": "test"},
            {"step": "清理旧代码", "tool": "edit_file"}
        ]
    }

    def match_template(self, task_description: str) -> str:
        """匹配任务模板"""

        task_lower = task_description.lower()

        # 关键词匹配
        if "修改" in task_description or "编辑" in task_description:
            return "修改文件"
        elif "添加" in task_description or "新增" in task_description:
            return "添加功能"
        elif "修复" in task_description or "bug" in task_lower or "错误" in task_lower:
            return "修复bug"
        elif "审查" in task_description or "review" in task_lower:
            return "代码审查"
        elif "重构" in task_description:
            return "重构"

        return None

    def decompose(self, task: str) -> List[dict]:
        """基于模板拆解任务"""

        template_name = self.match_template(task)

        if template_name and template_name in self.TEMPLATES:
            template = self.TEMPLATES[template_name]
            return [
                {
                    "description": step["step"],
                    "tool": step["tool"],
                    "template": template_name
                }
                for step in template
            ]

        # 无匹配模板，使用通用拆解
        return self._generic_decompose(task)

    def _generic_decompose(self, task: str) -> List[dict]:
        """通用拆解"""

        return [
            {"description": f"分析任务: {task}", "tool": "think"},
            {"description": f"执行任务: {task}", "tool": "execute"},
            {"description": f"验证结果", "tool": "verify"}
        ]
```

## 任务依赖管理

### 依赖分析

```python
class DependencyManager:
    """任务依赖管理器"""

    def __init__(self):
        self.dependencies: Dict[str, Set[str]] = {}
        self.dependents: Dict[str, Set[str]] = {}

    def add_dependency(self, task_id: str, depends_on: str):
        """添加依赖"""

        if task_id not in self.dependencies:
            self.dependencies[task_id] = set()
        self.dependencies[task_id].add(depends_on)

        if depends_on not in self.dependents:
            self.dependents[depends_on] = set()
        self.dependents[depends_on].add(task_id)

    def get_execution_order(self, task_ids: List[str]) -> List[str]:
        """获取执行顺序（拓扑排序）"""

        # 计算入度
        in_degree = {tid: 0 for tid in task_ids}
        for task_id in task_ids:
            for dep in self.dependencies.get(task_id, []):
                if dep in in_degree:
                    in_degree[task_id] += 1

        # 拓扑排序
        queue = [tid for tid in task_ids if in_degree[tid] == 0]
        result = []

        while queue:
            current = queue.pop(0)
            result.append(current)

            # 更新依赖者的入度
            for dependent in self.dependents.get(current, []):
                if dependent in in_degree:
                    in_degree[dependent] -= 1
                    if in_degree[dependent] == 0:
                        queue.append(dependent)

        return result

    def detect_cycles(self) -> List[List[str]]:
        """检测循环依赖"""
        # 使用 DFS 检测环
        cycles = []
        visited = set()
        rec_stack = set()

        def dfs(task_id, path):
            visited.add(task_id)
            rec_stack.add(task_id)

            for dep in self.dependencies.get(task_id, []):
                if dep not in visited:
                    if dfs(dep, path + [dep]):
                        return True
                elif dep in rec_stack:
                    # 发现环
                    cycle_start = path.index(dep)
                    cycles.append(path[cycle_start:] + [dep])

            rec_stack.remove(task_id)
            return False

        for task_id in self.dependencies:
            if task_id not in visited:
                dfs(task_id, [task_id])

        return cycles
```

### 自动依赖检测

```python
class AutoDependencyDetector:
    """自动依赖检测器"""

    def detect_dependencies(self, tasks: List[Task],
                           execution_results: dict) -> Dict[str, List[str]]:
        """检测任务间的依赖关系"""

        dependencies = {}

        for i, task in enumerate(tasks):
            deps = []

            # 检查之前的任务
            for j, prev_task in enumerate(tasks[:i]):
                if self._has_dependency(task, prev_task, execution_results):
                    deps.append(prev_task.id)

            dependencies[task.id] = deps

        return dependencies

    def _has_dependency(self, task: Task, prev_task: Task,
                       results: dict) -> bool:
        """判断是否有依赖"""

        task_desc = task.description.lower()
        prev_desc = prev_task.description.lower()
        prev_result = results.get(prev_task.id)

        # 1. 输出-输入依赖
        output_keywords = ["生成", "创建", "输出", "返回"]
        input_keywords = ["使用", "基于", "根据", "读取"]

        if any(kw in prev_desc for kw in output_keywords):
            if any(kw in task_desc for kw in input_keywords):
                return True

        # 2. 结果依赖
        if prev_task.status == TaskStatus.COMPLETED:
            if prev_result and isinstance(prev_result, str):
                if any(keyword in task_desc for keyword in prev_result[:100].split()):
                    return True

        return False
```

## 任务粒度控制

### 粒度原则

```python
class TaskGranularityController:
    """任务粒度控制器"""

    # 粒度阈值
    MAX_TASK_DURATION = 300  # 最大任务执行时间（秒）
    MAX_TASK_COMPLEXITY = 10  # 最大复杂度评分

    def __init__(self, tool_executor):
        self.executor = tool_executor

    def check_granularity(self, task: dict) -> tuple[bool, str]:
        """检查任务粒度是否合适"""

        description = task.get("description", "")

        # 1. 检查是否过于笼统
        vague_patterns = ["完成", "处理", "做好", "搞定"]
        if any(pattern in description for pattern in vague_patterns):
            return False, "任务描述过于笼统，需要更具体"

        # 2. 检查是否过长
        if len(description) > 200:
            return False, "任务描述过长，需要拆分"

        # 3. 检查是否包含多个动作
        action_count = self._count_actions(description)
        if action_count > 3:
            return False, f"任务包含 {action_count} 个动作，需要拆分为多个任务"

        return True, ""

    def _count_actions(self, description: str) -> int:
        """计算描述中的动作数量"""
        actions = ["修改", "添加", "删除", "创建", "更新", "修复", "检查", "验证"]
        count = sum(1 for action in actions if action in description)
        return max(count, 1)

    def split_if_needed(self, task: dict) -> List[dict]:
        """如果需要，拆分任务"""

        is_valid, reason = self.check_granularity(task)

        if is_valid:
            return [task]

        # 拆分任务
        return self._split_task(task)

    def _split_task(self, task: dict) -> List[dict]:
        """拆分任务"""
```

## 最佳实践

| 实践 | 说明 |
|------|------|
| 粒度适中 | 每个任务应该可以在 1-5 分钟内完成 |
| 明确目标 | 每个任务有明确的完成标准 |
| 独立可验证 | 任务可以独立验证，不依赖其他任务 |
| 处理依赖 | 明确任务间的依赖关系 |
| 动态调整 | 根据执行结果动态调整任务 |
| 深度限制 | 避免过度拆解导致碎片化 |
