# 03-状态机实现

状态机是管理 Agent 行为流程的核心机制，控制任务的执行、转换和终止。本文档详细介绍状态机的设计和实现。

## 状态机概述

### 什么是状态机

状态机是一种行为模型，包含有限的状态和状态之间的转换规则。

```
┌─────────────────────────────────────────────────────────────────┐
│                       Agent 状态机                              │
├─────────────────────────────────────────────────────────────────┤
│                                                                  │
│      ┌──────────┐                                               │
│      │  IDLE    │ ←────────── 初始化                             │
│      └────┬─────┘                                               │
│           │ start                                              │
│           ↓                                                    │
│      ┌──────────┐                                               │
│      │ PLANNING │  规划中                                       │
│      └────┬─────┘                                               │
│           │ plan_ready                                         │
│           ↓                                                    │
│      ┌──────────┐                                               │
│      │ EXECUTING│ ────→ ┌──────────┐                           │
│      └────┬─────┘       │ WAITING  │  等待工具执行              │
│           │ task_done  └────┬─────┘                            │
│           ↓                ↑                                   │
│      ┌──────────┐          │ wait_complete                    │
│      │COMPLETED │          │                                   │
│      └────┬─────┘          │ tool_result                       │
│           │                ↓                                   │
│           │            ┌──────────┐                             │
│           └──────────→ │  ERROR   │  错误处理                   │
│                        └──────────┘                             │
│                                                                  │
└─────────────────────────────────────────────────────────────────┘
```

## 状态定义

### Agent 状态

```python
from enum import Enum
from dataclasses import dataclass, field
from typing import Any, Optional, List
from datetime import datetime

class AgentState(Enum):
    """Agent 状态枚举"""
    IDLE = "idle"              # 空闲
    PLANNING = "planning"      # 规划中
    EXECUTING = "executing"    # 执行中
    WAITING = "waiting"        # 等待中（等待工具返回）
    COMPLETED = "completed"    # 完成
    ERROR = "error"            # 错误
    CANCELLED = "cancelled"    # 取消
    SUSPENDED = "suspended"     # 暂停

class EventType(Enum):
    """事件类型"""
    START = "start"
    PLAN_READY = "plan_ready"
    TASK_COMPLETE = "task_complete"
    TOOL_CALL = "tool_call"
    TOOL_RESULT = "tool_result"
    ERROR = "error"
    TIMEOUT = "timeout"
    CANCEL = "cancel"
    SUSPEND = "suspend"
    RESUME = "resume"

@dataclass
class AgentStatus:
    """Agent 状态信息"""
    state: AgentState
    current_task: Optional[str] = None
    task_history: List[dict] = field(default_factory=list)
    error_message: Optional[str] = None
    metadata: dict = field(default_factory=dict)
    started_at: Optional[datetime] = None
    last_updated: Optional[datetime] = None

@dataclass
class Event:
    """事件"""
    event_type: EventType
    data: Any = None
    timestamp: datetime = field(default_factory=datetime.now)
```

## 状态机实现

### 基础状态机

```python
from typing import Callable, Dict, List

class StateMachine:
    """状态机基类"""

    def __init__(self, initial_state: AgentState):
        self.current_state = initial_state
        self.state_history = [initial_state]
        self.listeners: Dict[AgentState, List[Callable]] = {
            state: [] for state in AgentState
        }

    def transition(self, new_state: AgentState, data: Any = None):
        """状态转换"""

        if not self._can_transition(self.current_state, new_state):
            raise ValueError(
                f"无效转换: {self.current_state.value} -> {new_state.value}"
            )

        old_state = self.current_state
        self.current_state = new_state
        self.state_history.append(new_state)

        # 触发监听器
        self._notify_listeners(new_state, data)

        print(f"状态转换: {old_state.value} -> {new_state.value}")

    def _can_transition(self, from_state: AgentState,
                       to_state: AgentState) -> bool:
        """检查是否可以转换"""

        # 定义转换规则
        valid_transitions = {
            AgentState.IDLE: [AgentState.PLANNING],
            AgentState.PLANNING: [AgentState.EXECUTING, AgentState.ERROR, AgentState.IDLE],
            AgentState.EXECUTING: [AgentState.WAITING, AgentState.COMPLETED,
                                   AgentState.ERROR, AgentState.SUSPENDED],
            AgentState.WAITING: [AgentState.EXECUTING, AgentState.ERROR],
            AgentState.COMPLETED: [AgentState.IDLE],
            AgentState.ERROR: [AgentState.IDLE, AgentState.PLANNING],
            AgentState.CANCELLED: [AgentState.IDLE],
            AgentState.SUSPENDED: [AgentState.EXECUTING, AgentState.CANCELLED],
        }

        return to_state in valid_transitions.get(from_state, [])

    def _notify_listeners(self, state: AgentState, data: Any):
        """通知状态监听器"""
        for listener in self.listeners.get(state, []):
            listener(data)

    def on(self, state: AgentState, callback: Callable):
        """注册状态监听器"""
        self.listeners[state].append(callback)
```

### Agent 状态机

```python
class AgentStateMachine(StateMachine):
    """Agent 状态机"""

    def __init__(self, agent_id: str):
        super().__init__(AgentState.IDLE)
        self.agent_id = agent_id
        self.current_plan = None
        self.current_task_index = 0
        self.execution_context = {}

    def start(self, goal: str, context: dict):
        """启动 Agent"""

        # 验证
        if self.current_state != AgentState.IDLE:
            raise ValueError("Agent 已经在运行中")

        # 初始化上下文
        self.execution_context = {
            "goal": goal,
            "context": context,
            "start_time": datetime.now()
        }

        # 转换到规划状态
        self.transition(AgentState.PLANNING, {"goal": goal})

    def plan_ready(self, plan: dict):
        """计划制定完成"""

        if self.current_state != AgentState.PLANNING:
            return

        self.current_plan = plan
        self.current_task_index = 0
        self.transition(AgentState.EXECUTING, plan)

    def execute_next_task(self, executor) -> Any:
        """执行下一个任务"""

        if self.current_state != AgentState.EXECUTING:
            return None

        if not self.current_plan:
            raise ValueError("没有可执行的计划")

        tasks = self.current_plan.get("tasks", [])
        if self.current_task_index >= len(tasks):
            # 所有任务完成
            self.transition(AgentState.COMPLETED)
            return None

        # 获取当前任务
        task = tasks[self.current_task_index]
        self.execution_context["current_task"] = task

        # 转换到等待状态（工具执行中）
        self.transition(AgentState.WAITING, task)

        # 执行任务
        return executor.execute(task)

    def task_complete(self, result: Any):
        """任务完成"""

        if self.current_state != AgentState.WAITING:
            return

        # 保存结果
        task = self.execution_context.get("current_task")
        if "results" not in self.execution_context:
            self.execution_context["results"] = []
        self.execution_context["results"].append({
            "task": task,
            "result": result,
            "completed_at": datetime.now()
        })

        # 移动到下一个任务
        self.current_task_index += 1

        if self.current_task_index >= len(self.current_plan.get("tasks", [])):
            # 所有任务完成
            self.transition(AgentState.COMPLETED)
        else:
            # 继续执行
            self.transition(AgentState.EXECUTING)

    def handle_error(self, error: Exception):
        """处理错误"""

        self.execution_context["error"] = {
            "message": str(error),
            "timestamp": datetime.now(),
            "state": self.current_state.value
        }

        self.transition(AgentState.ERROR, {"error": str(error)})

    def cancel(self):
        """取消执行"""

        self.transition(AgentState.CANCELLED)
        self._cleanup()

    def suspend(self):
        """暂停执行"""

        if self.current_state in [AgentState.EXECUTING, AgentState.WAITING]:
            self.transition(AgentState.SUSPENDED)

    def resume(self):
        """恢复执行"""

        if self.current_state == AgentState.SUSPENDED:
            self.transition(AgentState.EXECUTING)

    def _cleanup(self):
        """清理资源"""
        self.current_plan = None
        self.execution_context = {}
```

## 状态机使用示例

### Agent 循环

```python
class AgentLoop:
    """Agent 执行循环"""

    def __init__(self, state_machine: AgentStateMachine,
                 planner, executor, max_iterations: int = 100):
        self.state_machine = state_machine
        self.planner = planner
        self.executor = executor
        self.max_iterations = max_iterations
        self.iteration = 0

    async def run(self, goal: str, context: dict = None) -> dict:
        """运行 Agent"""

        context = context or {}
        self.iteration = 0

        # 1. 启动
        self.state_machine.start(goal, context)

        # 2. 规划
        if self.state_machine.current_state == AgentState.PLANNING:
            plan = await self.planner.create_plan(goal, context)
            self.state_machine.plan_ready(plan)

        # 3. 执行循环
        while self._should_continue():
            self.iteration += 1

            # 执行任务
            result = self.state_machine.execute_next_task(self.executor)

            # 等待工具执行完成
            if result:
                # 模拟等待工具返回
                tool_result = await self._wait_for_tool(result)
                self.state_machine.task_complete(tool_result)

            # 检查状态
            if self.state_machine.current_state == AgentState.COMPLETED:
                break

            if self.state_machine.current_state == AgentState.ERROR:
                break

        # 4. 返回结果
        return self._build_result()

    def _should_continue(self) -> bool:
        """判断是否继续执行"""

        if self.iteration >= self.max_iterations:
            print(f"达到最大迭代次数: {self.max_iterations}")
            return False

        if self.state_machine.current_state == AgentState.ERROR:
            return False

        if self.state_machine.current_state == AgentState.CANCELLED:
            return False

        if self.state_machine.current_state == AgentState.COMPLETED:
            return False

        return True

    async def _wait_for_tool(self, tool_call):
        """等待工具执行"""
        # 实现等待逻辑
        return {"status": "success", "data": None}

    def _build_result(self) -> dict:
        """构建结果"""

        context = self.state_machine.execution_context

        return {
            "status": self.state_machine.current_state.value,
            "goal": context.get("goal"),
            "iterations": self.iteration,
            "results": context.get("results", []),
            "error": context.get("error")
        }
```

### 状态持久化

```python
import json
from pathlib import Path

class StatePersistence:
    """状态持久化"""

    def __init__(self, storage_path: str):
        self.storage_path = Path(storage_path)
        self.storage_path.mkdir(parents=True, exist_ok=True)

    def save_state(self, agent_id: str, state_machine: AgentStateMachine):
        """保存状态"""

        state_data = {
            "agent_id": agent_id,
            "current_state": state_machine.current_state.value,
            "state_history": [s.value for s in state_machine.state_history],
            "current_plan": state_machine.current_plan,
            "current_task_index": state_machine.current_task_index,
            "execution_context": self._serialize_context(
                state_machine.execution_context
            ),
            "timestamp": datetime.now().isoformat()
        }

        file_path = self.storage_path / f"{agent_id}.json"
        with open(file_path, 'w', encoding='utf-8') as f:
            json.dump(state_data, f, ensure_ascii=False, indent=2)

    def load_state(self, agent_id: str) -> dict:
        """加载状态"""

        file_path = self.storage_path / f"{agent_id}.json"

        if not file_path.exists():
            return None

        with open(file_path, 'r', encoding='utf-8') as f:
            return json.load(f)

    def _serialize_context(self, context: dict) -> dict:
        """序列化上下文，移除不可序列化对象"""
        serialized = {}
        for key, value in context.items():
            try:
                json.dumps(value)
                serialized[key] = value
            except:
                serialized[key] = str(value)
        return serialized
```

### 断点续做

```python
class ResumableAgent:
    """可恢复的 Agent"""

    def __init__(self, state_machine: AgentStateMachine,
                 persistence: StatePersistence):
        self.state_machine = state_machine
        self.persistence = persistence

    def resume(self, agent_id: str, executor) -> dict:
        """恢复执行"""

        # 加载状态
        state_data = self.persistence.load_state(agent_id)

        if not state_data:
            raise ValueError(f"找不到 Agent 状态: {agent_id}")

        # 恢复状态机
        self.state_machine.current_state = AgentState(
            state_data["current_state"]
        )
        self.state_machine.current_plan = state_data.get("current_plan")
        self.state_machine.current_task_index = state_data.get(
            "current_task_index", 0
        )
        self.state_machine.execution_context = state_data.get(
            "execution_context", {}
        )

        # 继续执行
        return self._continue_execution(executor)

    def _continue_execution(self, executor) -> dict:
        """继续执行"""

        # 类似 AgentLoop 的执行逻辑
        while self.state_machine.current_state == AgentState.EXECUTING:
            result = self.state_machine.execute_next_task(executor)
            if result:
                tool_result = self._wait_for_tool(result)
                self.state_machine.task_complete(tool_result)

            if self.state_machine.current_state in [
                AgentState.COMPLETED, AgentState.ERROR
            ]:
                break

        return self._build_result()
```

## 状态监控与调试

### 状态监控

```python
class StateMonitor:
    """状态监控器"""

    def __init__(self):
        self.state_changes = []
        self.metrics = {
            "total_transitions": 0,
            "error_count": 0,
            "waiting_time": []
        }

    def record_transition(self, from_state: AgentState,
                         to_state: AgentState, data: Any):
        """记录状态转换"""

        self.state_changes.append({
            "from": from_state.value,
            "to": to_state.value,
            "data": data,
            "timestamp": datetime.now()
        })

        self.metrics["total_transitions"] += 1

        if to_state == AgentState.ERROR:
            self.metrics["error_count"] += 1

    def get_state_timeline(self) -> List[dict]:
        """获取状态时间线"""
        return self.state_changes

    def get_statistics(self) -> dict:
        """获取统计信息"""
        return {
            **self.metrics,
            "current_state": self.state_changes[-1]["to"] if self.state_changes else None,
            "avg_waiting_time": sum(self.metrics["waiting_time"]) /
                               len(self.metrics["waiting_time"]) if self.metrics["waiting_time"] else 0
        }
```
