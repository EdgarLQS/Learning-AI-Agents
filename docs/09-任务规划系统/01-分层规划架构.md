# 01-分层规划架构

分层规划将任务分解到不同的抽象层次，使 Agent 能够处理从简单到复杂的各种任务。本文档详细介绍战略层、战术层和执行层的架构设计。

## 分层规划概述

### 为什么需要分层

```
┌─────────────────────────────────────────────────────────────────┐
│                     分层规划的必要性                             │
├─────────────────────────────────────────────────────────────────┤
│                                                                  │
│  简单任务 ──────────────────────────────→ 一步到位               │
│  "计算 1+1" → 直接执行                                          │
│                                                                  │
│  中等任务 ──────────────────────────────→ 两层规划               │
│  "查找函数定义" → 搜索 → 读取文件                                │
│                                                                  │
│  复杂任务 ──────────────────────────────→ 三层规划               │
│  "重构这个模块" → 分析 → 规划 → 执行 → 验证                     │
│                                                                  │
│  ─────────────────────────────────────────────────────────────   │
│                                                                  │
│  不分层的问题：                                                  │
│  1. 规划步骤太多导致上下文爆炸                                   │
│  2. 难以处理任务间的依赖关系                                     │
│  3. 执行中难以调整计划                                           │
│                                                                  │
└─────────────────────────────────────────────────────────────────┘
```

### 三层架构设计

```python
from abc import ABC, abstractmethod
from dataclasses import dataclass, field
from typing import List, Dict, Any, Optional
from enum import Enum

class TaskStatus(Enum):
    """任务状态"""
    PENDING = "pending"
    IN_PROGRESS = "in_progress"
    COMPLETED = "completed"
    FAILED = "failed"
    BLOCKED = "blocked"

@dataclass
class Task:
    """任务单元"""
    id: str
    description: str
    status: TaskStatus = TaskStatus.PENDING
    depends_on: List[str] = field(default_factory=list)
    result: Any = None
    error: Optional[str] = None
    metadata: Dict = field(default_factory=dict)

@dataclass
class Plan:
    """执行计划"""
    id: str
    goal: str
    tasks: List[Task]
    current_task_index: int = 0
    metadata: Dict = field(default_factory=dict)
```

## 战略层 (Strategic Layer)

### 职责

战略层负责理解用户目标，制定高层次的计划，评估可行性。

```python
class StrategicPlanner:
    """战略层规划器"""

    def __init__(self, llm_client):
        self.llm = llm_client

    async def create_plan(self, goal: str, context: dict) -> Plan:
        """创建高层次计划"""

        # 1. 分析目标
        goal_analysis = await self._analyze_goal(goal, context)

        # 2. 生成计划大纲
        plan_outline = await self._generate_outline(goal, goal_analysis)

        # 3. 评估可行性
        feasibility = await self._assess_feasibility(plan_outline, context)

        if not feasibility["viable"]:
            # 不可行，返回建议
            return Plan(
                id=self._generate_id(),
                goal=goal,
                tasks=[],
                metadata={"reason": feasibility["reason"]}
            )

        # 4. 创建任务列表
        tasks = await self._create_tasks(plan_outline)

        return Plan(
            id=self._generate_id(),
            goal=goal,
            tasks=tasks,
            metadata={"analysis": goal_analysis}
        )

    async def _analyze_goal(self, goal: str, context: dict) -> dict:
        """分析目标"""
        prompt = f"""分析以下目标：

目标：{goal}

上下文信息：
{self._format_context(context)}

请分析：
1. 目标的类型（代码修改、信息查询、问题诊断等）
2. 涉及的代码范围
3. 预期的输出形式
4. 潜在的风险

请以 JSON 格式返回分析结果。"""

        response = await self.llm.chat.completions.create(
            model="gpt-4",
            messages=[{"role": "user", "content": prompt}]
        )

        return self._parse_json(response.choices[0].message.content)

    async def _generate_outline(self, goal: str, analysis: dict) -> dict:
        """生成计划大纲"""

        prompt = f"""基于以下分析，为目标生成高层次计划：

目标：{goal}
分析：{analysis}

生成计划大纲，包含：
1. 主要阶段（3-6个阶段）
2. 每个阶段的目标
3. 阶段之间的依赖关系

请以 JSON 格式返回。"""

        # 类似的处理...
        return {}

    async def _assess_feasibility(self, outline: dict, context: dict) -> dict:
        """评估可行性"""

        # 检查上下文是否足够
        # 检查目标是否明确
        # 检查是否有足够的信息

        return {"viable": True, "reason": None}

    async def _create_tasks(self, outline: dict) -> List[Task]:
        """根据大纲创建任务"""
        tasks = []

        for i, stage in enumerate(outline.get("stages", [])):
            task = Task(
                id=f"task_{i}",
                description=stage["description"],
                status=TaskStatus.PENDING,
                depends_on=stage.get("depends_on", [])
            )
            tasks.append(task)

        return tasks

    def _format_context(self, context: dict) -> str:
        return "\n".join(f"- {k}: {v}" for k, v in context.items())

    def _parse_json(self, text: str) -> dict:
        import json
        try:
            return json.loads(text)
        except:
            return {}

    def _generate_id(self) -> str:
        import uuid
        return str(uuid.uuid4())
```

## 战术层 (Tactical Layer)

### 职责

战术层负责将高层计划细化为具体的可执行步骤，管理任务依赖。

```python
class TacticalPlanner:
    """战术层规划器"""

    def __init__(self, tool_registry):
        self.tool_registry = tool_registry

    async def refine_plan(self, plan: Plan, execution_context: dict) -> Plan:
        """细化计划"""

        refined_tasks = []

        for task in plan.tasks:
            # 跳过已完成的任务
            if task.status == TaskStatus.COMPLETED:
                refined_tasks.append(task)
                continue

            # 细化每个任务
            refined = await self._refine_task(task, execution_context)

            # 如果任务可以进一步分解
            if refined.get("subtasks"):
                refined_tasks.extend(refined["subtasks"])
            else:
                refined_tasks.append(task)

        plan.tasks = refined_tasks
        return plan

    async def _refine_task(self, task: Task, context: dict) -> dict:
        """细化单个任务"""

        # 判断任务复杂度
        if self._is_complex(task):
            # 分解为子任务
            return await self._decompose_task(task, context)
        else:
            # 添加具体的执行步骤
            return await self._add_steps(task, context)

    def _is_complex(self, task: Task) -> bool:
        """判断任务是否复杂"""
        # 基于关键词或任务描述判断
        complex_keywords = ["重构", "修复", "多个", "重构", "全面"]
        return any(kw in task.description for kw in complex_keywords)

    async def _decompose_task(self, task: Task, context: dict) -> dict:
        """分解复杂任务"""

        prompt = f"""将以下任务分解为具体的子任务：

返回 JSON 格式：
{{
  "subtasks": [
    {{"id": "task_1", "description": "...", "depends_on": []}},
    ...
  ]
}}"""

        # 调用 LLM 获取分解结果
        # ...

        return {"subtasks": []}

    async def _add_steps(self, task: Task, context: dict) -> dict:
        """为任务添加执行步骤"""

        # 确定需要的工具
        needed_tools = self._identify_tools(task)

        # 添加步骤
        task.metadata["steps"] = [
            {"tool": tool, "purpose": "..."}
            for tool in needed_tools
        ]

        return {"subtasks": []}

    def _identify_tools(self, task: Task) -> List[str]:
        """识别任务需要的工具"""
        # 基于任务描述和关键词匹配
        tools = []

        if "搜索" in task.description or "查找" in task.description:
            tools.extend(["search", "grep", "glob"])

        if "读取" in task.description or "查看" in task.description:
            tools.append("read_file")

        if "修改" in task.description or "编辑" in task.description:
            tools.extend(["read_file", "edit_file"])

        if "执行" in task.description or "运行" in task.description:
            tools.append("bash")

        return list(set(tools))
```

## 执行层 (Execution Layer)

### 职责

执行层负责具体执行任务，验证结果，并处理错误。

```python
class ExecutionPlanner:
    """执行层规划器"""

    def __init__(self, tool_executor):
        self.executor = tool_executor

    async def execute_task(self, task: Task, context: dict) -> Task:
        """执行单个任务"""

        task.status = TaskStatus.IN_PROGRESS

        try:
            # 获取依赖任务的结果
            deps_results = self._get_dependency_results(task, context)

            # 执行任务
            result = await self._execute_with_tools(
                task, deps_results, context
            )

            task.result = result
            task.status = TaskStatus.COMPLETED

            return task

        except Exception as e:
            task.error = str(e)
            task.status = TaskStatus.FAILED
            return task

    def _get_dependency_results(self, task: Task, context: dict) -> dict:
        """获取依赖任务的结果"""
        results = {}

        for dep_id in task.depends_on:
            # 从 context 中获取依赖任务的结果
            task_obj = context.get(f"task_{dep_id}")
            if task_obj:
                results[dep_id] = task_obj.result

        return results

    async def _execute_with_tools(self, task: Task,
                                   deps_results: dict,
                                   context: dict) -> Any:
        """使用工具执行任务"""

        # 获取执行步骤
        steps = task.metadata.get("steps", [])

        if not steps:
            # 没有预设步骤，动态决定
            steps = await self._plan_steps(task, deps_results, context)

        # 顺序执行
        current_result = None
        for step in steps:
            tool = step["tool"]
            params = self._prepare_params(step, current_result, deps_results)

            result = await self.executor.execute(tool, params)
            current_result = result

            # 检查结果是否需要调整
            if self._needs_adjustment(result, task):
                steps = await self._adjust_plan(steps, result, context)

        return current_result

    async def _plan_steps(self, task: Task, deps: dict, context: dict) -> List[dict]:
        """动态规划执行步骤"""
        # 可以调用 LLM 决定步骤
        return []

    def _prepare_params(self, step: dict, prev_result: Any, deps: dict) -> dict:
        """准备工具参数"""
        params = step.get("params", {})

        # 填充变量
        for key, value in params.items():
            if isinstance(value, str) and value.startswith("$"):
                var_name = value[1:]
                if var_name == "prev_result":
                    params[key] = prev_result
                elif var_name in deps:
                    params[key] = deps[var_name]

        return params

    def _needs_adjustment(self, result: Any, task: Task) -> bool:
        """判断是否需要调整计划"""
        # 检查结果是否符合预期
        return False

    async def _adjust_plan(self, steps: List[dict],
                          result: Any, context: dict) -> List[dict]:
        """调整计划"""
        return steps
```

## 完整示例

### 任务规划流程

```python
class HierarchicalPlanner:
    """分层规划器"""

    def __init__(self, llm_client, tool_registry, tool_executor):
        self.strategic = StrategicPlanner(llm_client)
        self.tactical = TacticalPlanner(tool_registry)
        self.execution = ExecutionPlanner(tool_executor)

    async def plan_and_execute(self, goal: str, context: dict) -> Plan:
        """完整的规划和执行流程"""

        # 1. 战略层：创建计划
        print("📋 战略层：分析目标...")
        plan = await self.strategic.create_plan(goal, context)

        if not plan.tasks:
            print(f"⚠️ 计划不可行: {plan.metadata.get('reason')}")
            return plan

        print(f"   计划包含 {len(plan.tasks)} 个任务")

        # 2. 战术层：细化计划
        print("📋 战术层：细化任务...")
        plan = await self.tactical.refine_plan(plan, context)

        print(f"   细化后包含 {len(plan.tasks)} 个任务")

        # 3. 执行层：执行任务
        print("📋 执行层：执行任务...")

        for i, task in enumerate(plan.tasks):
            print(f"   [{i+1}/{len(plan.tasks)}] {task.description}")

            # 检查依赖
            if not self._can_execute(task, plan.tasks):
                task.status = TaskStatus.BLOCKED
                continue

            # 执行
            task = await self.execution.execute_task(task, context)

            # 更新上下文
            context[f"task_{task.id}"] = task

            if task.status == TaskStatus.FAILED:
                print(f"   ❌ 失败: {task.error}")
                # 可以选择停止或继续
                if not self._should_continue(task, plan):
                    break

        return plan

    def _can_execute(self, task: Task, all_tasks: List[Task]) -> bool:
        """检查任务是否可以执行"""
        for dep_id in task.depends_on:
            dep_task = next((t for t in all_tasks if t.id == dep_id), None)
            if not dep_task or dep_task.status != TaskStatus.COMPLETED:
                return False
        return True

    def _should_continue(self, failed_task: Task, plan: Plan) -> bool:
        """判断是否继续执行"""
        # 基于错误类型和配置决定
        return True
```
