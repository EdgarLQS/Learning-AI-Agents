# 01-短期记忆与上下文

短期记忆是 Agent 理解当前任务的基础，包括会话历史、工作缓存和上下文窗口管理。本文档详细介绍如何高效管理上下文。

## 上下文窗口概述

### 什么是上下文窗口

上下文窗口（Context Window）是 LLM 可以一次性处理的最大 token 数量。超出这个限制的内容将无法被模型"看到"。

```
┌─────────────────────────────────────────────────────────────────┐
│                      上下文窗口示意                              │
├─────────────────────────────────────────────────────────────────┤
│                                                                  │
│  ┌──────────────────────────────────────────────────────────┐   │
│  │                    Context Window (如 128K)              │   │
│  ├──────────────────────────────────────────────────────────┤   │
│  │                                                          │   │
│  │  ┌──────────┐ ┌──────────┐ ┌──────────┐ ┌──────────┐    │   │
│  │  │ System   │ │ User     │ │ Assistant│ │ User     │    │   │
│  │  │ Prompt   │ │ Message  │ │ Response │ │ Message  │    │   │
│  │  │          │ │          │ │          │ │ (最新)   │    │   │
│  │  └──────────┘ └──────────┘ └──────────┘ └──────────┘    │   │
│  │                                                          │   │
│  │  ←───────────────── Token 预算消耗 ←────────────────→   │   │
│  │                                                          │   │
│  └──────────────────────────────────────────────────────────┘   │
│                                                                  │
│  如果新消息导致超限，需要压缩或丢弃早期内容                        │
│                                                                  │
└─────────────────────────────────────────────────────────────────┘
```

### 主流模型的上下文窗口

| 模型 | 上下文窗口 | 标准输入 | 扩展输入 |

## 会话历史管理

### 滑动窗口策略

```python
from typing import List, Dict
from dataclasses import dataclass
import tiktoken

@dataclass
class Message:
    role: str  # "system", "user", "assistant"
    content: str
    token_count: int = 0

class ConversationHistory:
    """会话历史管理器"""

    def __init__(self, max_tokens: int = 100000, model: str = "gpt-4"):
        self.max_tokens = max_tokens
        self.messages: List[Message] = []
        self.encoding = tiktoken.get_encoding("cl100k_base")

    def add_message(self, role: str, content: str):
        """添加消息"""
        token_count = len(self.encoding.encode(content))
        message = Message(role=role, content=content, token_count=token_count)
        self.messages.append(message)

        # 自动裁剪
        self._ensure_limit()

    def get_context(self) -> List[Dict[str, str]]:
        """获取当前上下文"""
        return [{"role": m.role, "content": m.content} for m in self.messages]

    def _ensure_limit(self):
        """确保不超过 token 限制"""
        while self._total_tokens() > self.max_tokens and len(self.messages) > 1:
            # 移除最旧的消息（保留 system）
            if self.messages[1].role != "system":
                self.messages.pop(1)
            else:
                # 如果第二个是 system，尝试压缩它
                break

    def _total_tokens(self) -> int:
        """计算总 token 数"""
        return sum(m.token_count for m in self.messages)
```

```python
class LayeredMemory:
    """分层记忆管理"""

    def __init__(self, config: dict):
        # 第一层：核心信息（永远保留）

        self.core_token_limit = 2000

        # 第二层：重要上下文
        self.important_memory = []
        self.important_token_limit = 10000

        # 第三层：完整历史（按需加载）
        self.full_history = []
        self.full_history_token_limit = 80000

    def add(self, role: str, content: str):
        """添加记忆"""
        # 根据角色和内容决定存储层次
        message = {"role": role, "content": content}

        if role == "system" or self._is_critical(content):
            self.core_memory.append(message)
        elif self._is_important(content):
            self.important_memory.append(message)
        else:
            self.full_history.append(message)

        self._compress_if_needed()

    def get_context(self) -> List[Dict[str, str]]:
        """获取当前上下文"""
        context = list(self.core_memory)
```

### 摘要压缩策略

```python
import re

class SummarizationCompressor:
    """摘要压缩器"""

    def __init__(self, llm_client):
        self.llm = llm_client

    async def compress(self, messages: List[Message], target_tokens: int) -> List[Message]:
        """压缩消息列表"""

        if not messages:
            return messages

        # 1. 保留最近的几条消息
        recent_count = 5
        recent = messages[-recent_count:]

        # 2. 对历史消息进行摘要
        older = messages[:-recent_count]
        if not older:
            return messages

        # 3. 生成摘要
        summary = await self._summarize(older)

        # 4. 构建压缩后的消息
        compressed = [
            Message(
                role="system",
                content=f"以下是对之前对话的摘要：{summary}",
                token_count=len(summary) // 4  # 粗略估算
            )
        ]
        compressed.extend(recent)

        return compressed

    async def _summarize(self, messages: List[Message]) -> str:
        """使用 LLM 生成摘要"""
        # 构建摘要提示
        conversation = "\n".join(f"{m.role}: {m.content[:200]}..." for m in messages)

        prompt = f"""请简洁地总结以下对话的要点（不超过200字）：

{conversation}

摘要："""

        # 调用 LLM
        summary = await self.llm.chat.completions.create(
            model="gpt-4",
            messages=[{"role": "user", "content": prompt}],
            max_tokens=500
        )

        return summary.choices[0].message.content
```

```python
        """更新当前任务"""
        self.state.current_task = task

    def add_completed_step(self, step: str):
        """添加完成的步骤"""
        self.state.completed_steps.append(step)

    def record_tool_use(self, tool_name: str, params: dict, result: str):
        """记录工具使用"""
        self.tools_used.append({
            "tool": tool_name,
            "params": params,
            "result": result[:500],  # 限制结果长度
            "timestamp": datetime.now()
        })

    def add_finding(self, finding: str):
        """添加关键发现"""
        self.key_findings.append(finding)

    def get_summary(self) -> str:
        """获取状态摘要"""
        lines = ["## 当前状态"]

        if self.state.current_task:
            lines.append(f"**任务**: {self.state.current_task}")

        if self.state.completed_steps:
            lines.append(f"**已完成**: {len(self.state.completed_steps)} 步")
            for step in self.state.completed_steps[-3:]:
                lines.append(f"  - {step}")

        if self.state.pending_steps:
            lines.append(f"**待处理**: {len(self.state.pending_steps)} 项")

        if self.key_findings:
            lines.append(f"**发现**: {len(self.key_findings)} 条")
            for f in self.key_findings[-2:]:
                lines.append(f"  - {f[:100]}...")

        return "\n".join(lines)
```

## 窗口管理策略

### Token 预算分配

```python
class TokenBudget:
    """Token 预算管理器"""

    def __init__(self, total_budget: int):
        self.total_budget = total_budget
        self.reserved = {
            "system": 500,        # System prompt 预留
            "output": 2000,       # 输出预留
            "buffer": 500,        # 缓冲空间
        }

    @property
    def available_for_input(self) -> int:
        """可用于输入的 token"""
        reserved = sum(self.reserved.values())
        return self.total_budget - reserved

    def allocate(self, component: str, percentage: float) -> int:
        """为组件分配预算"""
        available = self.available_for_input
        return int(available * percentage)


# 使用示例
budget = TokenBudget(total_budget=100000)  # 100K 上下文
system_budget = budget.allocate("system", 0.05)      # 5%
history_budget = budget.allocate("history", 0.60)    # 60%
context_budget = budget.allocate("context", 0.25)    # 25%
retrieved_budget = budget.allocate("retrieved", 0.10) # 10%
```

### 动态窗口调整

```python
class DynamicWindowManager:
    """动态窗口管理器 - 根据任务动态调整上下文"""

    def __init__(self, llm_client):
        self.llm = llm_client

    async def adapt_window(self, task_description: str,
                          available_tools: List[str]) -> dict:
        """根据任务自适应窗口分配"""

        # 分析任务复杂度
        complexity = await self._analyze_complexity(task_description)

        if complexity == "simple":
            # 简单任务：更多依赖历史
            return {
                "history_weight": 0.7,
                "context_weight": 0.2,
                "retrieved_weight": 0.1
            }
        elif complexity == "medium":
            return {
                "history_weight": 0.5,
                "context_weight": 0.3,
                "retrieved_weight": 0.2
            }
        else:
            # 复杂任务：更多上下文和检索
            return {
                "history_weight": 0.3,
                "context_weight": 0.4,
                "retrieved_weight": 0.3
            }

    async def _analyze_complexity(self, task: str) -> str:
        """分析任务复杂度"""
        # 简化实现：基于关键词判断
        complex_keywords = ["重构", "设计", "分析", "架构", "多个文件"]

        if any(kw in task for kw in complex_keywords):
            return "complex"

        medium_keywords = ["修改", "修复", "添加", "实现"]
        if any(kw in task for kw in medium_keywords):
            return "medium"

        return "simple"
```

## 最佳实践

| 实践 | 说明 |
|------|------|
| 保留 System Prompt | System prompt 是核心指令，始终保留 |
| 优先保留最新消息 | 越近的消息越有价值 |
| 分层管理 | 核心/重要/普通分层次管理 |
| 定期摘要 | 定期将历史消息压缩为摘要 |
| 预留输出空间 | 为模型输出预留足够的 token |
| 动态调整 | 根据任务复杂度动态分配预算 |
