## 可扩展性设计

```python
import asyncio
from typing import List

class AgentPool:
    """Agent 池 - 水平扩展"""

    def __init__(self, config: dict):
        self.config = config
        self.agents = []
        self.max_agents = config.get("max_agents", 10)

    async def get_agent(self) -> "Agent":
        """获取可用的 Agent"""
        # 找空闲的
        for agent in self.agents:
            if not agent.busy:
                return agent

        # 如果未达上限，创建新的
        if len(self.agents) < self.max_agents:
            agent = await self._create_agent()
            self.agents.append(agent)
            return agent

        # 等待空闲
        return await self._wait_for_agent()

    async def _create_agent(self) -> "Agent":
        """创建新 Agent"""
        return Agent(self.config)

    async def _wait_for_agent(self) -> "Agent":
        """等待可用 Agent"""
        while True:
            await asyncio.sleep(1)
            for agent in self.agents:
                if not agent.busy:
                    return agent
```

## 容错设计

```python
class CircuitBreaker:
    """熔断器"""

    def __init__(self, failure_threshold=5, recovery_time=60):
        self.failure_threshold = failure_threshold
        self.recovery_time = recovery_time
        self.failures = 0
        self.last_failure_time = None
        self.state = "closed"

    def call(self, func, *args, **kwargs):
        """执行带熔断保护"""

        if self.state == "open":
            if time.time() - self.last_failure_time > self.recovery_time:
                self.state = "half_open"
            else:
                raise CircuitOpenError()

        try:
            result = func(*args, **kwargs)
            self._on_success()
            return result
        except Exception as e:
            self._on_failure()
            raise

    def _on_success(self):
        self.failures = 0
        self.state = "closed"

    def _on_failure(self):
        self.failures += 1
        self.last_failure_time = time.time()

        if self.failures >= self.failure_threshold:
            self.state = "open"
```

## 安全设计

```python
import hashlib
import hmac

class SecurityManager:
    """安全管理"""

    def __init__(self, secret_key: str):
        self.secret_key = secret_key

    def sign_request(self, data: dict) -> str:
        """签名请求"""
        message = json.dumps(data, sort_keys=True)
        signature = hmac.new(
            self.secret_key.encode(),
            message.encode(),
            hashlib.sha256
        ).hexdigest()
        return signature

    def verify_request(self, data: dict, signature: str) -> bool:
        """验证请求"""
        expected = self.sign_request(data)
        return hmac.compare_digest(expected, signature)
```

## 监控和指标

```python
from prometheus_client import Counter, Histogram, Gauge

# 指标定义
requests_total = Counter('agent_requests_total', 'Total requests')
request_duration = Histogram('agent_request_duration_seconds', 'Request duration')
active_agents = Gauge('agent_active', 'Number of active agents')

class MetricsCollector:
    """指标收集器"""

    def record_request(self, agent_id: str, duration: float):
        requests_total.inc()
        request_duration.observe(duration)

    def set_active_agents(self, count: int):
        active_agents.set(count)
```
