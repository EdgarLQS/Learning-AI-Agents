# 01-整体架构设计

整体架构设计是构建 Agent 系统的基础，需要考虑分层、模块划分和组件交互。本文档详细介绍分层架构设计和模块划分原则。

## 分层架构设计

### 为什么需要分层

```
┌─────────────────────────────────────────────────────────────────┐
│                       分层架构的优势                             │
├─────────────────────────────────────────────────────────────────┤
│                                                                  │
│  1. 关注点分离                                                  │
│     - UI 只需要关注用户体验                                       │
│     - Core 只需要关注业务逻辑                                     │
│     - 能力层只需要关注功能实现                                     │
│                                                                  │
│  2. 可维护性                                                    │
│     - 修改一层不影响其他层                                        │
│     - 易于定位问题                                               │
│                                                                  │
│  3. 可测试性                                                    │
│     - 每层可以独立测试                                           │
│     - 易于 Mock 依赖                                             │
│                                                                  │
│  4. 可扩展性                                                    │
│     - 易于添加新功能                                             │
│     - 支持水平扩展                                               │
│                                                                  │
└─────────────────────────────────────────────────────────────────┘
```

## 模块划分

### 1. 用户界面层

```python
class UILayer:
    """用户界面层 - 处理用户交互"""

    def __init__(self):
        self.handlers = []

    def register_handler(self, handler):
        """注册事件处理器"""
        self.handlers.append(handler)

    async def handle_input(self, user_input: str, context: dict) -> dict:
        """处理用户输入"""
        for handler in self.handlers:
            result = await handler.process(user_input, context)
            if result:
                return result
        return None
```

### 2. Agent Core 层

```python
class AgentCore:
    """Agent 核心 - 业务逻辑"""

    def __init__(self):
        self.planner = None      # 规划器
        self.executor = None     # 执行器
        self.state_machine = None # 状态机
        self.tool_scheduler = None # 工具调度器

    async def run(self, task: str, context: dict) -> dict:
        """运行 Agent"""
        # 1. 规划
        plan = await self.planner.create_plan(task, context)

        # 2. 执行
        result = await self.executor.execute(plan)

        return result
```

class Capabilities:

## 模块间通信

### 事件驱动架构

```python
from typing import Callable, Dict, List
from dataclasses import dataclass
from datetime import datetime

@dataclass
class Event:
    """事件"""
    type: str
    data: Any
    timestamp: datetime = field(default_factory=datetime.now)
    source: str = ""

class EventBus:
    """事件总线 - 模块间通信"""

    def __init__(self):
        self.subscribers: Dict[str, List[Callable]] = {}

    def subscribe(self, event_type: str, callback: Callable):
        """订阅事件"""
        if event_type not in self.subscribers:
            self.subscribers[event_type] = []
        self.subscribers[event_type].append(callback)

    def publish(self, event: Event):
        """发布事件"""
        for callback in self.subscribers.get(event.type, []):
            callback(event)

# 使用示例
event_bus = EventBus()

# Planner 订阅任务完成事件
event_bus.subscribe("task_completed", lambda e: print(f"任务完成: {e.data}"))

# 发布事件
event_bus.publish(Event(type="task_completed", data="数据分析完成"))
```

### 请求-响应模式

```python
from typing import Protocol, Any
import asyncio

class AgentComponent(Protocol):
    """Agent 组件接口"""

    async def process(self, request: dict) -> dict:
        ...

class RequestRouter:
    """请求路由器"""

    def __init__(self):
        self.routes: Dict[str, AgentComponent] = {}

    def register(self, path: str, component: AgentComponent):
        """注册路由"""
        self.routes[path] = component

    async def route(self, request: dict) -> dict:
        """路由请求"""
        path = request.get("path", "")
        component = self.routes.get(path)

        if not component:
            return {"error": "Not Found"}

        return await component.process(request)
```

## 依赖注入

```python
from typing import TypeVar, Generic

T = TypeVar('T')

class Container:
    """依赖注入容器"""

    def __init__(self):
        self.services: Dict[type, Any] = {}

    def register(self, interface: type, instance: Any):
        """注册服务"""
        self.services[interface] = instance

    def resolve(self, interface: type) -> Any:
        """解析服务"""
        return self.services.get(interface)

# 使用示例
container = Container()

# 注册服务
container.register(LLMClient, OpenAIClient())
container.register(VectorStore, ChromaStore())
container.register(AgentCore, AgentCore())

# 解析使用
llm = container.resolve(LLMClient)
```

## 完整系统初始化

```python
class AgentSystem:
    """完整的 Agent 系统"""

    def __init__(self, config: AgentConfig):
        self.config = config
        self.event_bus = EventBus()
        self.container = Container()

        # 初始化各层
        self._init_infrastructure()
        self._init_capabilities()
        self._init_core()
        self._init_ui()

    def _init_infrastructure(self):
        """初始化基础设施层"""
        # LLM 客户端
        llm_client = OpenAIClient(
            model=self.config.llm_model,
            temperature=self.config.llm_temperature
        )
        self.container.register(LLMClient, llm_client)

        # 向量数据库
        vector_db = ChromaStore()
        self.container.register(VectorStore, vector_db)

    def _init_capabilities(self):
        """初始化能力层"""
        llm = self.container.resolve(LLMClient)
        vector_db = self.container.resolve(VectorStore)

        # 工具系统
        tools = ToolSystem()
        self.container.register(ToolSystem, tools)

        # 记忆系统
        memory = MemorySystem(llm, vector_db)
        self.container.register(MemorySystem, memory)

        # 代码分析
        code_analysis = CodeAnalysisSystem()
        self.container.register(CodeAnalysisSystem, code_analysis)

    def _init_core(self):
        """初始化 Core 层"""
        # 获取依赖
        llm = self.container.resolve(LLMClient)
        tools = self.container.register(ToolSystem)
        memory = self.container.resolve(MemorySystem)

        # 创建核心
        agent_core = AgentCore(llm, tools, memory)
        self.container.register(AgentCore, agent_core)

    def _init_ui(self):
        """初始化 UI 层"""
        agent_core = self.container.resolve(AgentCore)

        cli = CLI(agent_core)
        self.container.register(CLI, cli)

    def run(self):
        """运行系统"""
        cli = self.container.resolve(CLI)
        cli.start()
```
