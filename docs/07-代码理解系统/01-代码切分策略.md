# 01-代码切分策略

代码切分（Chunking）是将大型代码库拆分为可管理单元的关键步骤。本文档详细介绍各种代码切分策略及其适用场景。

## 代码切分策略概述

```
┌─────────────────────────────────────────────────────────────────┐
│                    代码切分策略                                 │
├─────────────────────────────────────────────────────────────────┤
│                                                                  │
│  ┌──────────────┐   ┌──────────────┐   ┌──────────────┐        │
│  │   按文件     │   │   按函数     │   │   按语义    │        │
│  │ File-based  │   │ Function-based│   │Semantic     │        │
│  ├──────────────┤   ├──────────────┤   ├──────────────┤        │
│  │ 简单直接     │   │ 保留完整     │   │ 智能理解    │        │
│  │ 可能过大     │   │ 函数上下文   │   │ 需要解析    │        │
│  └──────────────┘   └──────────────┘   └──────────────┘        │

│                                                                  │
└─────────────────────────────────────────────────────────────────┘
```

## 1. 按文件切分

### 简单文件切分

```python
from pathlib import Path
from typing import List, Dict

class FileChunker:
    """按文件切分"""

    def __init__(self, max_tokens: int = 2000):
        self.max_tokens = max_tokens

    def chunk(self, file_path: str) -> List[Dict]:
        """将文件切分为 chunks"""

        path = Path(file_path)
        with open(path, 'r', encoding='utf-8') as f:
            content = f.read()

        tokens = self._estimate_tokens(content)

        if tokens <= self.max_tokens:
            # 文件较小，直接返回
            return [{
                "content": content,
                "file": str(path),
                "type": "file",
                "start_line": 1,
                "end_line": len(content.split('\n')),
            }]

        # 文件过大，按行数切分
        return self._chunk_by_lines(path, content)

    def _chunk_by_lines(self, path: Path, content: str) -> List[Dict]:
        """按行数切分"""
        lines = content.split('\n')
        chunks = []
        chunk_lines = []
        start_line = 1

        for i, line in enumerate(lines, 1):
            chunk_lines.append(line)

            # 每 200 行或遇到空行分隔作为一块
            if len(chunk_lines) >= 200 or (line.strip() == '' and len(chunk_lines) > 50):
                chunks.append({
                    "content": '\n'.join(chunk_lines),
                    "file": str(path),
                    "type": "file_chunk",
                    "start_line": start_line,
                    "end_line": i,
                })
                start_line = i + 1
                chunk_lines = []

        # 处理剩余行
        if chunk_lines:
            chunks.append({
                "content": '\n'.join(chunk_lines),
                "file": str(path),
                "type": "file_chunk",
                "start_line": start_line,
                "end_line": len(lines),
            })

        return chunks

    def _estimate_tokens(self, text: str) -> int:
        """简单 token 估算"""
        return len(text.split()) + len(text) // 4
```

### 文件类型过滤

```python
class FileChunkerWithFilter(FileChunker):
    """带过滤的文件切分器"""

    # 需要处理的文件类型
    CODE_EXTENSIONS = {
        '.py', '.js', '.ts', '.jsx', '.tsx',
        '.java', '.go', '.rs', '.cpp', '.c', '.h',
        '.rb', '.php', '.swift', '.kt', '.scala',
        '.html', '.css', '.vue', '.svelte',
        '.json', '.yaml', '.yml', '.toml',
        '.sql', '.sh', '.bash', '.zsh'
    }

    # 排除的文件
    EXCLUDE_PATTERNS = {
        '__pycache__', '.git', 'node_modules',
        'dist', 'build', '.next', '.nuxt',
        '.venv', 'venv', 'env', '.env',
        '.min.js', '.min.css', 'bundle.js'
    }

    def should_process(self, file_path: str) -> bool:
        """判断是否应该处理该文件"""

        path = Path(file_path)

        # 检查扩展名
        if path.suffix not in self.CODE_EXTENSIONS:
            return False

        # 检查排除模式
        for part in path.parts:
            if any(exclude in part for exclude in self.EXCLUDE_PATTERNS):
                return False

        return True
```

## 2. 按函数/类切分

### Python 函数切分

```python
import re
from typing import List, Dict, Optional

class PythonFunctionChunker:
    """Python 函数/类切分器"""

    # 函数和类定义的正则表达式
    FUNCTION_PATTERN = r'^(\s*(?:async\s+)?def\s+\w+|class\s+\w+)'
    IMPORT_PATTERN = r'^(\s*(?:import\s+\w+|from\s+\w+\s+import))'

    def __init__(self, include_imports: bool = True):
        self.include_imports = include_imports

    def chunk(self, file_path: str) -> List[Dict]:
        """按函数/类切分 Python 文件"""

        with open(file_path, 'r', encoding='utf-8') as f:
            content = f.read()

        lines = content.split('\n')
        chunks = []

        # 提取所有 import 语句
        imports = []
        if self.include_imports:
            for i, line in enumerate(lines):
                if re.match(self.IMPORT_PATTERN, line):
                    imports.append((i + 1, line))

        # 找到所有函数和类的定义位置
        definitions = []
        current_class = None

        for i, line in enumerate(lines):
            # 检测类定义
            class_match = re.match(r'^class\s+(\w+)', line)
            if class_match:
                current_class = class_match.group(1)
                definitions.append({
                    "line": i + 1,
                    "type": "class",
                    "name": current_class,
                })
                continue

            # 检测函数定义
            func_match = re.match(r'^\s*(async\s+)?def\s+(\w+)', line)
            if func_match:
                func_name = func_match.group(2)
                definitions.append({
                    "line": i + 1,
                    "type": "function",
                    "name": func_name,
                    "class": current_class,
                })

        # 为每个定义提取代码块
        for i, def_info in enumerate(definitions):
            start_line = def_info["line"]
            end_line = definitions[i + 1]["line"] - 1 if i + 1 < len(definitions) else len(lines)

            # 提取代码（包含 docstring）
            code_lines = lines[start_line - 1:end_line]
            code = '\n'.join(code_lines)

            # 查找相关的 import
            relevant_imports = self._find_relevant_imports(
                imports, def_info["name"], lines
            )

            chunks.append({
                "content": code,
                "file": file_path,
                "type": f"{def_info['type']}_def",
                "name": def_info["name"],
                "start_line": start_line,
                "end_line": end_line,
                "imports": relevant_imports,
            })

        return chunks

    def _find_relevant_imports(self, imports: List[tuple],
                               func_name: str,
                               all_lines: List[str]) -> List[str]:
        """查找相关 import"""
        relevant = []

        # 查找函数体内使用的 import
        func_lines = all_lines  # 简化处理
        import_names = {imp[1].split()[1] if 'import' in imp[1] else ''
                       for imp in imports}

        return relevant
```

### JavaScript/TypeScript 切分

```python
class JavaScriptChunker:
    """JavaScript/TypeScript 函数切分器"""

    # 匹配函数、类、箭头函数
    DEFINITION_PATTERNS = [
        r'^(export\s+)?(async\s+)?function\s+(\w+)',  # function foo() {}
        r'^(export\s+)?const\s+(\w+)\s*=\s*(async\s+)?\(',  # const foo = () => {}
        r'^(export\s+)?class\s+(\w+)',  # class Foo {}
        r'^(export\s+)(async\s+)?function\s*\*\s*(\w+)',  # function* foo() {}
    ]

    def chunk(self, file_path: str) -> List[Dict]:
        """切分 JS/TS 文件"""

        with open(file_path, 'r', encoding='utf-8') as f:
            content = f.read()

        # 简化实现：使用 AST 解析更准确
        return self._chunk_with_esprima(file_path, content)

    def _chunk_with_esprima(self, file_path: str, content: str) -> List[Dict]:
        """使用正则做简单切分"""

        chunks = []
        lines = content.split('\n')

        # 找到所有定义
        definitions = []
        for i, line in enumerate(lines):
            for pattern in self.DEFINITION_PATTERNS:
                match = re.match(pattern, line)
                if match:
                    name = match.group(3) or match.group(2)
                    def_type = "function" if "function" in pattern else "class"
                    definitions.append({
                        "line": i + 1,
                        "type": def_type,
                        "name": name
                    })

        # 提取代码块
        for i, def_info in enumerate(definitions):
            start = def_info["line"] - 1
            end = definitions[i + 1]["line"] - 1 if i + 1 < len(definitions) else len(lines)

            code = '\n'.join(lines[start:end])

            chunks.append({
                "content": code,
                "file": file_path,
                "type": f"{def_info['type']}_def",
                "name": def_info["name"],
                "start_line": def_info["line"],
                "end_line": end,
            })

        return chunks
```

## 3. 按语义切分

### 基于 AST 的智能切分

```python
class SemanticChunker:
    """基于语义的代码切分器"""

    def __init__(self, language: str = "python"):
        self.language = language

    def chunk(self, content: str, file_path: str = "") -> List[Dict]:
        """语义切分"""

        if self.language == "python":
            return self._chunk_python(content, file_path)
        elif self.language in ("javascript", "typescript"):
            return self._chunk_js(content, file_path)
        else:
            # 其他语言回退到行数切分
            return self._chunk_by_lines(content, file_path)

    def _chunk_python(self, content: str, file_path: str) -> List[Dict]:
        """Python 语义切分"""
        try:
            import ast

            tree = ast.parse(content)
            chunks = []

            for node in ast.walk(tree):
                if isinstance(node, (ast.FunctionDef, ast.AsyncFunctionDef, ast.ClassDef)):
                    # 获取节点的位置信息
                    start_line = node.lineno
                    end_line = node.end_lineno

                    # 提取代码
                    lines = content.split('\n')
                    code = '\n'.join(lines[start_line - 1:end_line])

                    chunks.append({
                        "content": code,
                        "file": file_path,
                        "type": "class" if isinstance(node, ast.ClassDef) else "function",
                        "name": node.name,
                        "start_line": start_line,
                        "end_line": end_line,
                    })

            return chunks
        except SyntaxError:
            return self._chunk_by_lines(content, file_path)

    def _chunk_by_lines(self, content: str, file_path: str,
                       lines_per_chunk: int = 100) -> List[Dict]:
        """回退：按行数切分"""
        lines = content.split('\n')
        chunks = []

        for i in range(0, len(lines), lines_per_chunk):
            chunk_lines = lines[i:i + lines_per_chunk]
            chunks.append({
                "content": '\n'.join(chunk_lines),
                "file": file_path,
                "type": "lines",
                "start_line": i + 1,
                "end_line": min(i + lines_per_chunk, len(lines)),
            })

        return chunks
```

## 4. 混合策略

### 自适应切分器

```python
class AdaptiveChunker:
    """自适应代码切分器 - 根据文件特点选择策略"""

    def __init__(self):
        self.python_chunker = PythonFunctionChunker()
        self.js_chunker = JavaScriptChunker()
        self.semantic_chunker = SemanticChunker()

    def chunk(self, file_path: str) -> List[Dict]:
        """自动选择切分策略"""

        path = Path(file_path)
        suffix = path.suffix

        # 获取文件大小
        size = path.stat().st_size

        # 小文件：使用语义切分
        if size < 5000:
            return self._semantic_chunk(file_path, suffix)

        # 中等文件：使用函数切分
        elif size < 50000:
            return self._function_chunk(file_path, suffix)

        # 大文件：使用分层切分
        else:
            return self._hybrid_chunk(file_path, suffix)

    def _semantic_chunk(self, file_path: str, suffix: str) -> List[Dict]:
        """语义切分"""
        return self.semantic_chunk(file_path)

    def _function_chunk(self, file_path: str, suffix: str) -> List[Dict]:
        """函数级别切分"""
        if suffix == '.py':
            return self.python_chunker.chunk(file_path)
        elif suffix in ('.js', '.ts', '.jsx', '.tsx'):
            return self.js_chunker.chunk(file_path)
        else:
            return self.semantic_chunk(file_path)

    def _hybrid_chunk(self, file_path: str, suffix: str) -> List[Dict]:
        """混合切分：顶层 + 详细"""

        chunks = []

        # 1. 首先提取顶层定义（文件级别的概述）
        top_level = self._extract_top_level(file_path, suffix)
        chunks.extend(top_level)

        # 2. 然后按需添加详细实现
        detailed = self._function_chunk(file_path, suffix)
        chunks.extend(detailed)

        return chunks

    def _extract_top_level(self, file_path: str, suffix: str) -> List[Dict]:
        """提取顶层结构"""
        with open(file_path, 'r', encoding='utf-8') as f:
            content = f.read()

        # 简化：提取类名和函数名列表
        import re

        if suffix == '.py':
            classes = re.findall(r'^class\s+(\w+)', content, re.MULTILINE)
            funcs = re.findall(r'^def\s+(\w+)', content, re.MULTILINE)
        else:
            classes = re.findall(r'^class\s+(\w+)', content, re.MULTILINE)
            funcs = re.findall(r'^(?:export\s+)?function\s+(\w+)', content, re.MULTILINE)

        return [{
            "content": f"Classes: {classes}\nFunctions: {funcs}",
            "file": file_path,
            "type": "overview",
            "name": "file_overview",
            "start_line": 1,
            "end_line": 1,
        }]
```

## 代码切分最佳实践

### 策略选择指南

| 文件特征 | 推荐策略 | 原因 |
|----------|----------|------|
| < 5KB | 语义切分 | 保持完整语义 |
| 5-50KB | 函数切分 | 保持函数完整性 |
| > 50KB | 混合策略 | 平衡粒度和效率 |
| 单文件库 | 语义切分 | 需要完整上下文 |
| 大型项目 | 分层切分 | 先看结构再看细节 |

### 切分效果验证

```python
def validate_chunks(chunks: List[CodeChunk], original_content: str) -> dict:
    """验证切分效果"""

    # 1. 检查内容完整性
    total_length = sum(len(c.content) for c in chunks)
    original_length = len(original_content)

    # 2. 检查行号连续性
    line_gaps = []
    for i in range(len(chunks) - 1):
        current_end = chunks[i].end_line
        next_start = chunks[i + 1].start_line
        if current_end < next_start - 1:
            line_gaps.append((current_end, next_start))

    return {
        "total_chunks": len(chunks),
        "content_coverage": total_length / original_length if original_length else 0,
        "line_gaps": line_gaps,
        "avg_chunk_size": total_length / len(chunks) if chunks else 0,
    }
```
