# 03-工具安全性

工具安全性是 Agent 系统的生命线。不当的工具暴露可能导致数据泄露、文件损坏甚至服务器被控制。本文档详细介绍权限控制、危险操作隔离和审计日志。

## 1. 权限控制

### 基于角色的权限模型

```python
from enum import Enum
from typing import Set

class Permission(Enum):
    READ_FILE = "read_file"
```

### 工具级别权限控制

```python
class ToolPermissionChecker:
    def __init__(self):
        # 工具权限映射
        self.tool_permissions = {
            "read_file": Permission.READ_FILE,
            "write_file": Permission.WRITE_FILE,
            "delete_file": Permission.DELETE_FILE,
            "execute_command": Permission.EXECUTE_COMMAND,
            "web_search": Permission.NETWORK_ACCESS,
            "send_email": Permission.NETWORK_ACCESS,
        }

    def check_permission(self, tool_name: str, user_role: Role) -> bool:
        """检查用户角色是否有权使用工具"""
        required_permission = self.tool_permissions.get(tool_name)
        if not required_permission:
            return False  # 未知工具默认拒绝

        allowed_permissions = ROLE_PERMISSIONS.get(user_role, set())
        return required_permission in allowed_permissions
```

### 路径权限控制

```python
import os
from pathlib import Path

class PathPermissionChecker:
    def __init__(self, allowed_paths: List[str], denied_paths: List[str]):
        self.allowed_paths = [Path(p).resolve() for p in allowed_paths]
        self.denied_paths = [Path(p).resolve() for p in denied_paths]

    def check_path(self, file_path: str) -> bool:
        """检查路径是否在允许范围内"""
        try:
            resolved = Path(file_path).resolve()

            # 检查是否在拒绝列表中
            for denied in self.denied_paths:
                if str(resolved).startswith(str(denied)):
                    return False

            # 检查是否在允许列表中
            if not self.allowed_paths:
                return True

            for allowed in self.allowed_paths:
                if str(resolved).startswith(str(allowed)):
                    return True

            return False
        except Exception:
            return False  # 解析失败默认拒绝
```

## 2. 危险操作隔离

### 敏感操作确认

```python
class SensitiveOperationGuard:
    """敏感操作需要二次确认"""

    SENSITIVE_OPERATIONS = {
        "delete_file": {
            "patterns": ["*.conf", "*.key", "*.pem", "*.env"],
            "require_confirmation": True,
        },
        "execute_command": {
            "patterns": ["rm -rf", "dd ", "mkfs", ">:", "| sh", "bash -c"],
            "require_confirmation": True,
        },
        "write_file": {
            "patterns": ["*.sh", "*.exe", "*.dll"],
            "require_confirmation": True,
        },
    }

    def is_sensitive(self, operation: str, params: dict) -> bool:
        """判断操作是否敏感"""
        config = self.SENSITIVE_OPERATIONS.get(operation, {})
        if not config:
            return False

        # 检查参数中的敏感模式
        for key, value in params.items():
            if isinstance(value, str):
                for pattern in config.get("patterns", []):
                    if pattern in value:
                        return True

        return False

    def get_warning_message(self, operation: str, params: dict) -> str:
        """生成警告消息"""
        return f"""⚠️ 敏感操作警告

您即将执行: {operation}
参数: {params}

此操作可能不可逆，请确认是否继续？"""
```

### 沙箱执行环境

```python
import subprocess
import tempfile
import os

class SandboxExecutor:
    """沙箱执行器，限制工具执行范围"""

    def __init__(self, working_dir: str, max_execution_time: int = 30):
        self.working_dir = Path(working_dir).resolve()
        self.max_execution_time = max_execution_time

    def execute_command(self, command: str) -> CommandResult:
        """在沙箱中执行命令"""

        # 1. 命令黑名单检查
        if self._is_dangerous_command(command):
            raise SecurityError("禁止执行危险命令")

        # 2. 限制工作目录
        cmd = f"cd {self.working_dir} && {command}"

        # 3. 设置资源限制
        try:
            result = subprocess.run(
                cmd,
                shell=True,
                capture_output=True,
                text=True,
                timeout=self.max_execution_time,
                cwd=str(self.working_dir),
                env=self._get_limited_env(),
            )
            return CommandResult(
                success=result.returncode == 0,
                stdout=result.stdout,
                stderr=result.stderr,
                returncode=result.returncode,
            )
        except subprocess.TimeoutExpired:
            raise SecurityError("命令执行超时")

    def _is_dangerous_command(self, command: str) -> bool:
        """检查是否为危险命令"""
        dangerous_patterns = [
            r"rm\s+-rf\s+/(?:\s|$)",  # 删除根目录
            r"dd\s+if=",               # 直接磁盘操作
            r">\s*/dev/",              # 写入设备文件
            r"mkfs",                   # 格式化
            r":\(\)\{",                # Fork 炸弹
            r"curl.*\|\s*sh",          # 远程脚本执行
            r"wget.*\|\s*sh",
        ]
        return any(re.search(p, command) for p in dangerous_patterns)

    def _get_limited_env(self) -> dict:
        """获取受限的环境变量"""
        safe_env = os.environ.copy()
        # 移除敏感环境变量
        sensitive_vars = ["AWS_SECRET_ACCESS_KEY", "API_KEY", "TOKEN"]
        for var in sensitive_vars:
            safe_env.pop(var, None)
        return safe_env
```

### 文件操作保护

```python
class FileOperationGuard:
    """文件操作保护"""

    def __init__(self, project_root: Path):
        self.project_root = project_root.resolve()
        self.backup_dir = self.project_root / ".agent_backups"

    def pre_write_check(self, file_path: str) -> BackupResult:
        """写入前检查并备份"""
        path = Path(file_path).resolve()

        # 检查是否在项目目录内
        if not str(path).startswith(str(self.project_root)):
            raise SecurityError(f"禁止访问项目目录外的文件: {path}")

        # 如果文件存在，先备份
        if path.exists():
            backup_path = self._create_backup(path)
            return BackupResult(backed_up=True, backup_path=backup_path)

        return BackupResult(backed_up=False)

    def _create_backup(self, file_path: Path) -> Path:
        """创建文件备份"""
        self.backup_dir.mkdir(exist_ok=True)
        timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
        backup_name = f"{file_path.name}.{timestamp}.bak"
        backup_path = self.backup_dir / backup_name

        import shutil
        shutil.copy2(file_path, backup_path)
        return backup_path
```

## 3. 审计日志

### 审计日志结构

```python
from dataclasses import dataclass, field
from datetime import datetime
from typing import Optional, Any
import json

@dataclass
class AuditLog:
    timestamp: datetime = field(default_factory=datetime.now)
    user_id: str = ""
    session_id: str = ""
    tool_name: str = ""
    parameters: dict = field(default_factory=dict)
    result: Optional[str] = None
    success: bool = False
    error: Optional[str] = None
    execution_time_ms: float = 0.0
    ip_address: Optional[str] = None
    user_agent: Optional[str] = None

    def to_dict(self) -> dict:
        return {
            "timestamp": self.timestamp.isoformat(),
            "user_id": self.user_id,
            "session_id": self.session_id,
            "tool_name": self.tool_name,
            "parameters": self._sanitize_params(self.parameters),
            "result": self.result[:1000] if self.result else None,  # 限制长度
            "success": self.success,
            "error": self.error,
            "execution_time_ms": self.execution_time_ms,
            "ip_address": self.ip_address,
        }

    def _sanitize_params(self, params: dict) -> dict:
        """清理敏感参数"""
        sensitive_keys = {"password", "token", "api_key", "secret"}
        return {k: "***" if k.lower() in sensitive_keys else v
                for k, v in params.items()}
```

```python
    async def monitor(self, audit_log: AuditLog):
        """监控并检测异常"""

        # 1. 检测频繁操作
        key = f"{audit_log.user_id}:{audit_log.tool_name}"
```

## 4. 综合安全策略

### 安全配置示例

```python
class AgentSecurityConfig:
    """Agent 安全配置"""

    def __init__(self):
        # 1. 路径限制
        self.path_checker = PathPermissionChecker(
            allowed_paths=["/project/src", "/project/tests"],
            denied_paths=["/project/.git", "/project/secrets"]
        )

        # 2. 工具权限
        self.permission_checker = ToolPermissionChecker()

        # 3. 敏感操作保护
        self.sensitive_guard = SensitiveOperationGuard()

        # 4. 沙箱执行
        self.sandbox = SandboxExecutor("/project")

        # 5. 文件操作保护
        self.file_guard = FileOperationGuard(Path("/project"))

        # 6. 审计日志
        self.audit_logger = AuditLogger("/var/log/agent")

        # 7. 安全监控
        self.security_monitor = SecurityMonitor()

    async def execute_tool(self, tool_name: str, params: dict,
                          user_role: Role, user_id: str) -> ToolResult:
        """安全执行工具"""

        # 1. 权限检查
        if not self.permission_checker.check_permission(tool_name, user_role):
            await self._log_and_alert(user_id, tool_name, "权限不足")
            return ToolResult(success=False, error="权限不足")

        # 2. 参数安全检查
        if "path" in params:
            if not self.path_checker.check_path(params["path"]):
                await self._log_and_alert(user_id, tool_name, "路径被拒绝")
                return ToolResult(success=False, error="路径不在允许范围内")

        # 3. 敏感操作确认
        if self.sensitive_guard.is_sensitive(tool_name, params):
            # 这里可以要求用户确认
            pass

        # 4. 执行前备份（如适用）
        if tool_name in ["write_file", "edit_file"]:
            backup_result = self.file_guard.pre_write_check(params.get("path", ""))
            params["_backup_path"] = backup_result.backup_path

        # 5. 执行工具
        start_time = time.time()
        result = await self._execute_tool(tool_name, params)
        execution_time = (time.time() - start_time) * 1000

        # 6. 记录审计日志
        await self._log_execution(user_id, tool_name, params, result, execution_time)

        return result

    async def _log_and_alert(self, user_id: str, tool_name: str, reason: str):
        """记录安全事件并告警"""
        print(f"🚨 安全拦截: 用户 {user_id} 尝试使用 {tool_name}: {reason}")
        # 告警逻辑

    async def _log_execution(self, user_id: str, tool_name: str,
                           params: dict, result: ToolResult, exec_time: float):
        """记录执行日志"""
        audit_log = AuditLog(
            user_id=user_id,
            tool_name=tool_name,
            parameters=params,
            result=result.result,
            success=result.success,
            error=result.error,
            execution_time_ms=exec_time
        )
        self.audit_logger.log(audit_log)
        await self.security_monitor.monitor(audit_log)
```

## 安全检查清单

| 检查项 | 说明 |
|--------|------|
| 权限最小化 | 只授予必要权限，不使用管理员权限运行 |
| 路径隔离 | 限制 Agent 只能访问指定目录 |
| 命令黑名单 | 禁止执行危险命令 |
| 操作确认 | 敏感操作需要二次确认 |
| 审计日志 | 记录所有工具调用 |
| 实时监控 | 检测异常行为并告警 |
| 备份机制 | 修改前先备份原文件 |
| 超时限制 | 防止命令无限执行 |
