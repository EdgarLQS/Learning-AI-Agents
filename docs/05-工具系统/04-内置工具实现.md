## 1. 文件操作工具

### 读取文件

```python
from pathlib import Path
from typing import Optional

class ReadFileTool:
    """读取文件内容"""

    name = "read_file"
    description = "读取指定文件的内容。支持部分读取（指定行号范围）和全部读取。"

    parameters = {
        "type": "object",
        "properties": {
            "path": {
                "type": "string",
                "description": "文件路径"
            },
            "offset": {
                "type": "integer",
                "description": "起始行号（1-indexed）",
                "default": 1
            },
            "limit": {
                "type": "integer",
                "description": "读取行数",
                "default": 100
            }
        },
        "required": ["path"]
    }

    def execute(self, path: str, offset: int = 1, limit: int = 100) -> str:
        """执行读取"""
        file_path = Path(path)

        if not file_path.exists():
            raise FileNotFoundError(f"文件不存在: {path}")

        with open(file_path, 'r', encoding='utf-8') as f:
            lines = f.readlines()

        # 转换为 1-indexed
        start = offset - 1
        end = start + limit

        selected_lines = lines[start:end]
        total_lines = len(lines)

        # 构建结果
        result = []
        for i, line in enumerate(selected_lines, start=offset):
            result.append(f"{i:4d}| {line.rstrip()}")

        response = "\n".join(result)

        # 添加元信息
        if end < total_lines:
            response += f"\n\n... (共 {total_lines} 行，已显示 {offset}-{end} 行) ..."
        else:
            response += f"\n\n--- (共 {total_lines} 行) ---"

        return response
```

### 编辑文件

```python
import re
from typing import Tuple

class EditFileTool:
    """编辑文件内容"""

    name = "edit_file"
    description = "对文件进行精确编辑，支持替换、插入、删除操作"

    parameters = {
        "type": "object",
        "properties": {
            "path": {
                "type": "string",
                "description": "文件路径"
            },
            "operation": {
                "type": "string",
                "enum": ["replace", "insert", "delete"],
                "description": "操作类型"
            },
            "old_string": {
                "type": "string",
                "description": "要替换的字符串（用于 replace 和 delete）"
            },
            "new_string": {
                "type": "string",
                "description": "要插入或替换成的字符串（用于 replace 和 insert）"
            },
            "replace_all": {
                "type": "boolean",
                "description": "是否替换所有匹配项",
                "default": False
            }
        },
        "required": ["path", "operation"]
    }

    def execute(self, path: str, operation: str,
                 old_string: str = None, new_string: str = None,
                 replace_all: bool = False) -> str:
        """执行编辑"""

        file_path = Path(path)
        if not file_path.exists():
            raise FileNotFoundError(f"文件不存在: {path}")

        # 读取原内容
        with open(file_path, 'r', encoding='utf-8') as f:
            content = f.read()

        # 执行操作
        if operation == "replace":
            if replace_all:
                new_content = content.replace(old_string, new_string)
            else:
                new_content = content.replace(old_string, new_string, 1)
            change_count = 1 if old_string in content else 0

        elif operation == "insert":
            # 在指定位置插入
            lines = content.split('\n')
            insert_idx = self._find_insert_index(lines, old_string)
            lines.insert(insert_idx, new_string)
            new_content = '\n'.join(lines)
            change_count = 1

        elif operation == "delete":
            new_content = content.replace(old_string, '', 1 if not replace_all else -1)
            change_count = content.count(old_string)

        else:
            raise ValueError(f"未知操作: {operation}")

        # 写回文件
        with open(file_path, 'w', encoding='utf-8') as f:
            f.write(new_content)

        return f"已完成 {operation} 操作，修改 {change_count} 处"

    def _find_insert_index(self, lines: list, after: str) -> int:
        """找到插入位置"""
        for i, line in enumerate(lines):
            if after in line:
                return i + 1
        return len(lines)
```

```python
class ListDirectoryTool:
    """列出目录内容"""

    name = "list_directory"
    description = "列出指定目录中的文件和子目录"

    parameters = {
        "type": "object",
        "properties": {
            "path": {
                "type": "string",
                "description": "目录路径"
            },
            "all": {
                "type": "boolean",
                "description": "是否显示隐藏文件",
                "default": False
            },
            "recursive": {
                "type": "boolean",
                "description": "是否递归列出子目录",
                "default": False
            }
        },
        "required": ["path"]
    }

    def execute(self, path: str, all: bool = False, recursive: bool = False) -> str:
        """执行列出"""
        dir_path = Path(path)

        if not dir_path.exists():
            raise FileNotFoundError(f"目录不存在: {path}")

        if not dir_path.is_dir():
            raise NotADirectoryError(f"不是目录: {path}")

        results = []

        if recursive:
            for item in dir_path.rglob("*"):
                if not all and item.name.startswith('.'):
                    continue
                rel_path = item.relative_to(dir_path)
                prefix = "📁 " if item.is_dir() else "📄 "
                results.append(f"{prefix}{rel_path}")
        else:
            for item in dir_path.iterdir():
                if not all and item.name.startswith('.'):
                    continue
                prefix = "📁 " if item.is_dir() else "📄 "
                size = f" ({item.stat().st_size} bytes)" if item.is_file() else ""
                results.append(f"{prefix}{item.name}{size}")

        return "\n".join(results) if results else "目录为空"
```

### Grep 搜索

```python
import re
from typing import List, Tuple
from pathlib import Path

class GrepTool:
    """代码搜索工具"""

    name = "grep"
    description = "在文件中搜索指定的文本或正则表达式"

    parameters = {
        "type": "object",
        "properties": {
            "pattern": {
                "type": "string",
                "description": "搜索模式（支持正则表达式）"
            },
            "path": {
                "type": "string",
                "description": "搜索路径（文件或目录）"
            },
            "glob": {
                "type": "string",
                "description": "文件过滤模式，如 *.py",
                "default": "*"
            },
            "ignore_case": {
                "type": "boolean",
                "description": "是否忽略大小写",
                "default": False
            },
            "line_numbers": {
                "type": "boolean",
                "description": "是否显示行号",
                "default": True
            },
            "max_results": {
                "type": "integer",
                "description": "最大结果数",
                "default": 50
            }
        },
        "required": ["pattern", "path"]
    }

    def execute(self, pattern: str, path: str, glob: str = "*",
                ignore_case: bool = False, line_numbers: bool = True,
                max_results: int = 50) -> str:
        """执行搜索"""

        flags = re.IGNORECASE if ignore_case else 0
        regex = re.compile(pattern, flags)

        results = []
        path_obj = Path(path)

        # 遍历文件
        for file_path in path_obj.rglob(glob) if path_obj.is_dir() else [path_obj]:
            if not file_path.is_file():
                continue

            try:
                with open(file_path, 'r', encoding='utf-8') as f:
                    for line_no, line in enumerate(f, 1):
                        if regex.search(line):
                            prefix = f"{file_path}:{line_no}:" if line_numbers else ""
                            results.append(f"{prefix}{line.rstrip()}")

                            if len(results) >= max_results:
                                break
            except Exception:
                continue

            if len(results) >= max_results:
                break

        if not results:
            return f"未找到匹配: {pattern}"

        output = "\n".join(results)
        if len(results) >= max_results:
            output += f"\n... (结果已截断，仅显示前 {max_results} 条) ..."

        return output
```

### 查找符号定义

```python
class FindDefinitionTool:
    """查找符号定义位置"""

    name = "find_definitions"
    description = "在代码中查找函数、类、变量的定义位置"

    parameters = {
        "type": "object",
        "properties": {
            "symbol": {
                "type": "string",
                "description": "要查找的符号名称"
            },
            "path": {
                "type": "string",
                "description": "搜索路径",
                "default": "."
            },
            "symbol_type": {
                "type": "string",
                "enum": ["function", "class", "variable", "all"],
                "description": "符号类型",
                "default": "all"
            }
        },
        "required": ["symbol"]
    }

    # 符号定义模式（简化版）
    DEFINITION_PATTERNS = {
        "function": [
            r"^(?:async\s+)?def\s+(\w+)",  # Python
            r"^function\s+(\w+)",           # JavaScript
            r"^fn\s+(\w+)",                 # Rust
            r"^func\s+(\w+)",               # Go
        ],
        "class": [
            r"^class\s+(\w+)",              # Python, JavaScript, Java
        ],
        "variable": [
            r"^(?:const|let|var)\s+(\w+)",  # JavaScript
            r"^(\w+)\s*=",                   # 变量赋值
        ]
    }

    def execute(self, symbol: str, path: str = ".",
                symbol_type: str = "all") -> str:
        """执行查找"""

        results = []
        root = Path(path)

        types = [symbol_type] if symbol_type != "all" else ["function", "class"]

        for file_path in root.rglob("*.py"):
            for type_ in types:
                matches = self._search_file(file_path, symbol, type_)
                results.extend(matches)

        if not results:
            return f"未找到符号 '{symbol}' 的定义"

        return "\n".join(results[:20])

    def _search_file(self, file_path: Path, symbol: str, symbol_type: str):
        """在文件中搜索符号"""
        matches = []
        patterns = self.DEFINITION_PATTERNS.get(symbol_type, [])

        try:
            with open(file_path, 'r', encoding='utf-8') as f:
                for line_no, line in enumerate(f, 1):
                    for pattern in patterns:
                        if re.search(pattern, line):
                            if symbol in line:
                                matches.append(f"{file_path}:{line_no}: {line.strip()}")
        except Exception:
            pass

        return matches
```

```python
import subprocess
import asyncio
from typing import Optional

class BashTool:
    """Shell 命令执行工具"""

    name = "bash"
    description = "执行 Shell 命令。用于运行程序、脚本或系统命令。"

    parameters = {
        "type": "object",
        "properties": {
            "command": {
                "type": "string",
                "description": "要执行的命令"
            },
            "timeout": {
                "type": "integer",
                "description": "超时时间（秒）",
                "default": 30
            },
            "description": {
                "type": "string",
                "description": "命令说明（可选）"
            }
        },
        "required": ["command"]
    }

    def execute(self, command: str, timeout: int = 30,
                description: str = None) -> str:
        """执行命令"""

        try:
            result = subprocess.run(
                command,
                shell=True,
                capture_output=True,
                text=True,
                timeout=timeout,
                cwd=None  # 可以限制工作目录
            )

            output = []

            if description:
                output.append(f"$ {description}")
            output.append(f"$ {command}")

            if result.stdout:
                output.append(result.stdout)

            if result.stderr:
                output.append(f"[stderr]: {result.stderr}")

            if result.returncode != 0:
                output.append(f"[exit code: {result.returncode}]")

            return "\n".join(output)

        except subprocess.TimeoutExpired:
            return f"命令执行超时（{timeout}秒）"
        except Exception as e:
            return f"命令执行失败: {str(e)}"
```

```python
        """检测测试框架"""
        if Path("pytest.ini").exists() or Path("pyproject.toml").exists():
            return "pytest"
        elif Path("package.json").exists():
            return "jest"
        return "pytest"

    def _run_pytest(self, path: str, verbose: bool) -> str:
        """运行 pytest"""
        v = "-v" if verbose else ""
        cmd = f"pytest {path} {v}"
```

```python
        if engine == "ddg":

        elif engine == "google":
            return self._search_google(query, num_results)

        return "不支持的搜索引擎"

    def _search_duckduckgo(self, query: str, num_results: int) -> str:
        """DuckDuckGo 搜索"""
        # 使用非官方 API
        url = "https://api.duckduckgo.com/"
        params = {
            "q": query,
            "format": "json",
            "no_html": 1,
            "skip_disambig": 1
        }

        try:
            response = requests.get(url, params=params, timeout=10)
            data = response.json()

            results = []
            for item in data.get("RelatedTopics", [])[:num_results]:
                if "Text" in item:
                    results.append(f"- {item['Text']}")

            return "\n".join(results) if results else "未找到结果"
        except Exception as e:
            return f"搜索失败: {str(e)}"
```

```python
import requests
from bs4 import BeautifulSoup

class WebFetchTool:
    """网页抓取工具"""

    name = "fetch_url"
    description = "获取指定 URL 的内容"

    parameters = {
        "type": "object",
        "properties": {
            "url": {
                "type": "string",
                "description": "要抓取的 URL"
            },
            "selector": {
                "type": "string",
                "description": "CSS 选择器（可选）"
            },
            "max_length": {
                "type": "integer",
                "description": "最大字符数",
                "default": 5000
            }
        },
        "required": ["url"]
    }

    def execute(self, url: str, selector: str = None,
                max_length: int = 5000) -> str:
        """抓取网页"""

        try:
            response = requests.get(url, timeout=10)
            response.raise_for_status()

            soup = BeautifulSoup(response.text, 'html.parser')

            # 提取内容
            if selector:
                content = soup.select(selector)
                text = "\n".join(c.get_text(strip=True) for c in content)
            else:
                # 提取纯文本
                for script in soup(["script", "style"]):
                    script.decompose()
                text = soup.get_text(separator="\n", strip=True)

            # 截断
            if len(text) > max_length:
                text = text[:max_length] + "\n... (内容已截断)"

            return text

        except requests.RequestException as e:
            return f"抓取失败: {str(e)}"
```
