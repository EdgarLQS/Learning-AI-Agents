## 1. 工具选择与匹配

### 基于描述的匹配

```python
class ToolMatcher:
    def __init__(self, tools: List[Tool]):
        self.tools = tools

    def match(self, user_query: str) -> List[Tool]:
        """根据用户查询匹配最相关的工具"""
        # 1. 计算查询与每个工具描述的相似度
        scores = []
        for tool in self.tools:
            score = self._calculate_relevance(user_query, tool.description)
            scores.append((tool, score))

        # 2. 返回相关性高于阈值的工具
        relevant_tools = [t for t, s in scores if s > 0.3]
        return sorted(relevant_tools, key=lambda t: t.score, reverse=True)

    def _calculate_relevance(self, query: str, description: str) -> float:
        # 使用 embedding 或关键词匹配计算相关性
        # 这里简化为关键词重叠
        query_words = set(query.lower().split())
        desc_words = set(description.lower().split())
        return len(query_words & desc_words) / len(query_words | desc_words)
```

### 基于上下文的动态选择

```python
class ContextAwareToolSelector:
    def select(self, conversation: Conversation, available_tools: List[Tool]) -> List[Tool]:
        """根据对话上下文动态选择工具"""

        # 1. 分析最近几轮的意图
        recent_intents = self._analyze_recent_intents(conversation)

        # 2. 根据意图过滤工具
        selected = []
        for tool in available_tools:
            if self._tool_matches_intent(tool, recent_intents):
                selected.append(tool)

        # 3. 如果没有匹配，按相似度选择
        if not selected:
            last_message = conversation.last_message
            selected = self._match_by_similarity(last_message, available_tools)

        return selected[:5]  # 限制返回数量

    def _tool_matches_intent(self, tool: Tool, intents: List[str]) -> bool:
        """检查工具是否匹配意图"""
        tool_keywords = tool.description.lower().split()
        return any(intent in tool_keywords for intent in intents)
```

## 2. 参数解析与校验

### 参数提取

```python
class ParameterParser:
    def parse(self, tool_call: dict, schema: dict) -> dict:
        """解析工具调用参数"""
        raw_params = tool_call.get("function", {}).get("arguments", {})

        # 处理字符串格式的 JSON
        if isinstance(raw_params, str):
            raw_params = json.loads(raw_params)

        # 类型转换和校验
        validated = {}
        properties = schema.get("parameters", {}).get("properties", {})

        for name, spec in properties.items():
            value = raw_params.get(name)

            if value is None:
                if name in schema.get("parameters", {}).get("required", []):
                    raise ParameterError(f"缺少必需参数: {name}")
                continue

            # 类型转换
            validated[name] = self._convert_type(value, spec["type"])

        return validated

    def _convert_type(self, value, expected_type: str):
        """类型转换"""
        if expected_type == "integer":
            return int(value)
        elif expected_type == "number":
            return float(value)
        elif expected_type == "boolean":
            return bool(value)
        return value
```

### 参数校验

```python
from typing import Any
import re

class ParameterValidator:
    def validate(self, params: dict, schema: dict) -> Tuple[bool, List[str]]:
        """校验参数是否符合 schema"""
        errors = []
        properties = schema.get("properties", {})

        for name, spec in properties.items():
            value = params.get(name)

            # 检查必填参数
            if value is None:
                if name in schema.get("required", []):
                    errors.append(f"缺少必需参数: {name}")
                continue

            # 类型校验
            if not self._check_type(value, spec["type"]):
                errors.append(f"参数 {name} 类型错误，期望 {spec['type']}")

            # 范围校验
            if "minimum" in spec and value < spec["minimum"]:
                errors.append(f"参数 {name} 小于最小值 {spec['minimum']}")
            if "maximum" in spec and value > spec["maximum"]:
                errors.append(f"参数 {name} 大于最大值 {spec['maximum']}")

            # 枚举校验
            if "enum" in spec and value not in spec["enum"]:
                errors.append(f"参数 {name} 不在允许的值范围内: {spec['enum']}")

            # 模式校验（正则）
            if "pattern" in spec and isinstance(value, str):
                if not re.match(spec["pattern"], value):
                    errors.append(f"参数 {name} 不符合模式: {spec['pattern']}")

        return len(errors) == 0, errors

    def _check_type(self, value: Any, expected_type: str) -> bool:
        """检查类型是否匹配"""
        type_map = {
            "string": str,
            "integer": int,
            "number": (int, float),
            "boolean": bool,
            "array": list,
            "object": dict
        }
        expected = type_map.get(expected_type)
        return isinstance(value, expected) if expected else True
```

## 3. 工具执行

### 执行器实现

```python
import asyncio
from typing import Any, Optional

class ToolExecutor:
    def __init__(self, max_retries: int = 3, timeout: int = 30):
        self.max_retries = max_retries
        self.timeout = timeout
        self.tool_registry = {}

    def register(self, name: str, func: callable, schema: dict):
        """注册工具"""
        self.tool_registry[name] = {
            "function": func,
            "schema": schema
        }

    async def execute(self, tool_name: str, parameters: dict) -> ToolResult:
        """执行工具并返回结果"""

        if tool_name not in self.tool_registry:
            return ToolResult(
                success=False,
                error=f"未知工具: {tool_name}",
                result=None
            )

        tool = self.tool_registry[tool_name]
        func = tool["function"]

        # 参数校验
        validator = ParameterValidator()
        is_valid, errors = validator.validate(parameters, tool["schema"])
        if not is_valid:
            return ToolResult(success=False, error=f"参数错误: {errors}", result=None)

        # 执行（带重试）
        last_error = None
        for attempt in range(self.max_retries):
            try:
                result = await self._execute_with_timeout(func, parameters)
                return ToolResult(success=True, result=result, error=None)
            except Exception as e:
                last_error = e
                if self._is_retryable(e):
                    await asyncio.sleep(2 ** attempt)  # 指数退避
                    continue
                break

        return ToolResult(success=False, error=str(last_error), result=None)

    async def _execute_with_timeout(self, func: callable, params: dict) -> Any:
        """带超时的执行"""
        if asyncio.iscoroutinefunction(func):
            return await asyncio.wait_for(func(**params), timeout=self.timeout)
        else:
            return func(**params)

    def _is_retryable(self, error: Exception) -> bool:
        """判断错误是否可重试"""
        retryable_errors = (TimeoutError, ConnectionError)
        return isinstance(error, retryable_errors)
```

## 4. 结果处理

### 结果解析与格式化

```python
@dataclass
class ToolResult:
    success: bool
    result: Any
    error: Optional[str] = None
    execution_time: float = 0.0

class ResultFormatter:
    def format(self, result: ToolResult) -> str:
        """将工具结果格式化为字符串，供 LLM 处理"""

        if not result.success:
            return f"工具执行失败: {result.error}"

        # 格式化不同类型的结果
        if result.result is None:
            return "操作完成，无返回结果"

        if isinstance(result.result, str):
            return result.result

        if isinstance(result.result, dict):
            return self._format_dict(result.result)

        if isinstance(result.result, list):
            return self._format_list(result.result)

        return str(result.result)

    def _format_dict(self, data: dict, indent: int = 0) -> str:
        """格式化字典"""
        lines = []
        for key, value in data.items():
            if isinstance(value, (dict, list)):
                lines.append(f"{'  ' * indent}{key}:")
                lines.append(self._format_nested(value, indent + 1))
            else:
                lines.append(f"{'  ' * indent}{key}: {value}")
        return "\n".join(lines)

    def _format_list(self, items: list) -> str:
        """格式化列表"""
        if not items:
            return "空列表"

        if all(isinstance(i, str) for i in items):
            return "\n".join(f"- {item}" for item in items)

        return "\n".join(str(item) for item in items[:10])  # 限制显示数量
```

### 结果截断

```python
class ResultTruncator:
    MAX_LENGTH = 8000  # 最大字符数

    def truncate(self, result: str) -> str:
        """截断过长的结果"""
        if len(result) <= self.MAX_LENGTH:
            return result

        # 保留开头和结尾
        head = result[:self.MAX_LENGTH // 2]
        tail = result[-self.MAX_LENGTH // 2:]

        return f"""{head}
... (结果已截断，原始长度 {len(result)} 字符) ...
{tail}"""
```

## 5. 错误处理与重试

### 错误分类

```python
from enum import Enum

class ErrorType(Enum):
    PARAMETER_ERROR = "参数错误"
    TIMEOUT_ERROR = "执行超时"
    PERMISSION_ERROR = "权限不足"
    NOT_FOUND_ERROR = "资源不存在"
    SYSTEM_ERROR = "系统错误"
    NETWORK_ERROR = "网络错误"

class ErrorHandler:
    def classify(self, error: Exception) -> ErrorType:
        """分类错误类型"""
        error_msg = str(error).lower()

        if "parameter" in error_msg or "缺少参数" in error_msg:
            return ErrorType.PARAMETER_ERROR
        elif "timeout" in error_msg or "超时" in error_msg:
            return ErrorType.TIMEOUT_ERROR
        elif "permission" in error_msg or "权限" in error_msg:
            return ErrorType.PERMISSION_ERROR
        elif "not found" in error_msg or "不存在" in error_msg:
            return ErrorType.NOT_FOUND_ERROR
        elif "network" in error_msg or "连接" in error_msg:
            return ErrorType.NETWORK_ERROR
        else:
            return ErrorType.SYSTEM_ERROR

    def should_retry(self, error_type: ErrorType) -> bool:
        """判断是否应该重试"""
        return error_type in [
            ErrorType.TIMEOUT_ERROR,
            ErrorType.NETWORK_ERROR,
            ErrorType.SYSTEM_ERROR
        ]
```

### 重试策略

```python
class RetryStrategy:
    def __init__(self, max_attempts: int = 3):
        self.max_attempts = max_attempts

    async def execute_with_retry(self, func: callable) -> Any:
        """执行带重试的操作"""
        last_error = None

        for attempt in range(self.max_attempts):
            try:
                return await func()
            except Exception as e:
                last_error = e

                if not self._should_retry(e):
                    raise

                # 指数退避
                wait_time = 2 ** attempt
                await asyncio.sleep(wait_time)

        raise last_error

    def _should_retry(self, error: Exception) -> bool:
        # 业务逻辑决定是否重试
        return True
```

## 6. 熔断机制

```python
from collections import deque
import time

class CircuitBreaker:
    def __init__(self, failure_threshold: int = 5, recovery_time: int = 60):
        self.failure_threshold = failure_threshold
        self.recovery_time = recovery_time
        self.failures = deque()
        self.state = "closed"  # closed, open, half_open

    def call(self, func: callable, *args, **kwargs):
        """熔断保护调用"""
        if self.state == "open":
            if time.time() - self.failures[0] > self.recovery_time:
                self.state = "half_open"
            else:
                raise CircuitOpenError("熔断器已打开")

        try:
            result = func(*args, **kwargs)
            if self.state == "half_open":
                self.state = "closed"
                self.failures.clear()
            return result
        except Exception as e:
            self.failures.append(time.time())
            if len(self.failures) >= self.failure_threshold:
                self.state = "open"
            raise
```

async def execute_tool_call(agent, tool_call: dict) -> str:
    """完整的工具调用流程"""

    # 1. 解析工具调用
    tool_name = tool_call["function"]["name"]

    # 2. 获取工具定义并校验参数
    tool = get_tool(tool_name)

        return f"参数错误: {', '.join(errors)}"

    # 3. 执行工具
    executor = ToolExecutor()
    result = await executor.execute(tool_name, raw_args)

    # 4. 格式化结果

    return truncator.truncate(formatted)
