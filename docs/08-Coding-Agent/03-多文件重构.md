## 依赖分析

### 导入关系分析

```python
class DependencyAnalyzer:
    """依赖分析器"""

    def analyze_imports(self, project_path: str) -> dict:
        """分析项目导入关系"""

        import os
        import ast

        imports = {}  # file -> [imported_files]

        for root, dirs, files in os.walk(project_path):
            for file in files:
                if file.endswith('.py'):
                    path = os.path.join(root, file)
                    imports[path] = self._extract_imports(path)

        # 构建依赖图
        return self._build_dependency_graph(imports)

    def _extract_imports(self, file_path: str) -> list:
        """提取导入"""
        try:
            with open(file_path, 'r') as f:
                tree = ast.parse(f.read())

            imports = []
            for node in ast.walk(tree):
                if isinstance(node, ast.Import):
                    for alias in node.names:
                        imports.append(alias.name)
                elif isinstance(node, ast.ImportFrom):
                    if node.module:
                        imports.append(node.module)

            return imports
        except:
            return []
```

## 回退机制

```python
class RefactorRollback:
    """重构回退"""

    def __init__(self):
        self.backups = []

    def backup(self, files: list):
        """备份文件"""
        for file in files:
            with open(file, 'r') as f:
                self.backups.append({
                    "file": file,
                    "content": f.read()
                })

    def rollback(self):
        """回退"""
        for backup in self.backups:
            with open(backup["file"], 'w') as f:
                f.write(backup["content"])
```
