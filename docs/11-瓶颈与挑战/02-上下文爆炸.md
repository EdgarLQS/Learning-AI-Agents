    def __init__(self, max_tokens: int = 80000):
        self.max_tokens = max_tokens

    def optimize_context(self, messages: list) -> list:
        """优化上下文"""

        # 计算可用空间

### 2. 上下文摘要

```python
class ContextSummarizer:
    """上下文摘要器"""

    def __init__(self, llm_client):
        self.llm = llm_client
        self.summary_interval = 10  # 每 10 轮生成摘要

    def should_summarize(self, message_count: int) -> bool:
        """判断是否需要摘要"""
        return message_count % self.summary_interval == 0

    async def summarize(self, messages: list) -> str:
        """生成摘要"""

        prompt = f"""总结以下对话的要点（不超过200字）：

{self._format_messages(messages)}

摘要:"""

        response = await self.llm.chat.completions.create(
            model="gpt-4",
            messages=[{"role": "user", "content": prompt}]
        )

        return response.choices[0].message.content
```
