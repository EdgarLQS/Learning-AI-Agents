## 语言特定处理

```python
class LanguageSpecificHandler:
    """语言特定处理器"""

    HANDLERS = {
        "python": PythonHandler(),
        "javascript": JavaScriptHandler(),
        "typescript": TypeScriptHandler(),
        "cpp": CppHandler(),
        "rust": RustHandler(),
        "go": GoHandler(),
    }

    def get_handler(self, language: str) -> BaseHandler:
        """获取语言处理器"""
        return self.HANDLERS.get(language, GenericHandler())
```

## 多语言项目处理

```python
class MultiLanguageProject:
    """多语言项目处理"""

    def __init__(self):
        self.language_handlers = {}

    def detect_languages(self, project_path: str) -> list:
        """检测项目语言"""
        languages = set()

        for file in Path(project_path).rglob("*"):
            if file.is_file():
                ext = file.suffix
                lang = self._ext_to_language(ext)
                if lang:
                    languages.add(lang)

        return list(languages)

    def _ext_to_language(self, ext: str) -> str:
        """扩展名到语言"""
        mapping = {
            ".py": "python",
            ".js": "javascript",
            ".ts": "typescript",
            ".cpp": "cpp",
            ".rs": "rust",
            ".go": "go",
        }
        return mapping.get(ext)
```
